3use kernel_info::boot::{BootPixelFormat, BootPixelMasks, FramebufferInfo};
4use log::{error, info};
5use uefi::boot::ScopedProtocol;
6use uefi::proto::console::gop::{GraphicsOutput, PixelFormat};
7use uefi::{Status, boot};

9/// Fetch an optimal framebuffer for the Kernel.
10pub fn get_framebuffer() -> Result<FramebufferInfo, Status> {
11    info!("Obtaining Graphics Output Protocol (GOP)");
12    let mut gop = match get_gop() {
13        Ok(gop) => gop,
14        Err(e) => {
15            error!("Failed to get GOP: {e:?}");
16            return Err(Status::UNSUPPORTED);
17        }
18    };
19
20    let mode = gop.current_mode_info();
21    let (framebuffer_width, framebuffer_height) = mode.resolution();
22
23    let (framebuffer_format, framebuffer_masks) = match mode.pixel_format() {
24        PixelFormat::Rgb => (
25            BootPixelFormat::Rgb,
26            BootPixelMasks {
27                red_mask: 0,
28                green_mask: 0,
29                blue_mask: 0,
30                alpha_mask: 0,
31            },
32        ),
33        PixelFormat::Bgr => (
34            BootPixelFormat::Bgr,
35            BootPixelMasks {
36                red_mask: 0,
37                green_mask: 0,
38                blue_mask: 0,
39                alpha_mask: 0,
40            },
41        ),
42        PixelFormat::Bitmask if mode.pixel_bitmask().is_some() => {
43            let mask = mode.pixel_bitmask().unwrap();
44            (
45                BootPixelFormat::Bitmask,
46                BootPixelMasks {
47                    red_mask: mask.red,
48                    green_mask: mask.green,
49                    blue_mask: mask.blue,
50                    alpha_mask: mask.reserved,
51                },
52            )
53        }
54        PixelFormat::BltOnly | PixelFormat::Bitmask => {
55            error!("Unsupported pixel format: Bitmask");
56            return Err(Status::UNSUPPORTED);
57        }
58    };
59
60    let mut fb = gop.frame_buffer();
61    let framebuffer_ptr = fb.as_mut_ptr();
62    let framebuffer_size = fb.size();
63    let framebuffer_stride = mode.stride();
64
65    let fb = FramebufferInfo {
66        framebuffer_ptr: framebuffer_ptr as u64,
67        framebuffer_size: framebuffer_size as u64,
68        framebuffer_width: framebuffer_width as u64,
69        framebuffer_height: framebuffer_height as u64,
70        framebuffer_stride: framebuffer_stride as u64,
71        framebuffer_format,
72        framebuffer_masks,
73    };
74
75    Ok(fb)
76}

78/// Fetch the Graphics Output Protocol (GOP).
79fn get_gop() -> Result<ScopedProtocol<GraphicsOutput>, uefi::Error> {
80    let handle = boot::get_handle_for_protocol::<GraphicsOutput>().map_err(|e| {
81        error!("Failed to get GOP handle: {e:?}");
82        uefi::Error::new(Status::ABORTED, ())
83    })?;
84
85    let gop = boot::open_protocol_exclusive::<GraphicsOutput>(handle).map_err(|e| {
86        error!("Failed to open GOP exclusively: {e:?}");
87        uefi::Error::new(Status::ABORTED, ())
88    })?;
89    Ok(gop)
90}
