1use kernel_qemu::qemu_trace;
2use log::{LevelFilter, Log, Metadata, Record, SetLoggerError};
3
4pub struct UefiLogger {
5    max_level: LevelFilter,
6    boot_services_available: bool,
7}
8
9impl UefiLogger {
10    #[must_use]
11    pub const fn new(max_level: LevelFilter) -> Self {
12        Self {
13            max_level,
14            boot_services_available: true,
15        }
16    }
17
18    /// Call this once during early init.
19    #[allow(
20        static_mut_refs,
21        clippy::missing_errors_doc,
22        clippy::missing_panics_doc
23    )]
24    pub fn init(self) -> Result<&'static mut Self, SetLoggerError> {
25        // SAFETY: log::set_logger expects &'static Log. Use a leaked boxed (or a static) in kernels.
26        // For no-alloc, we'll use a static.
27        static mut LOGGER: Option<UefiLogger> = None;
28
29        // move self into static
30        unsafe {
31            LOGGER = Some(self);
32            // set_logger requires &'static dyn Log
33            log::set_logger(LOGGER.as_ref().unwrap() as &'static dyn Log)?;
34        }
35        log::set_max_level(LevelFilter::Trace);
36        unsafe { Ok(LOGGER.as_mut().expect("initialized")) }
37    }
38
39    pub const fn exit_boot_services(&mut self) {
40        self.boot_services_available = false;
41    }
42}
43
44impl Log for UefiLogger {
45    fn enabled(&self, metadata: &Metadata) -> bool {
46        metadata.level() <= self.max_level
47    }
48
49    fn log(&self, record: &Record) {
50        if !self.enabled(record.metadata()) {
51            return;
52        }
53
54        // Format: "[LEVEL] target: message\n"
55        // Keep allocations out â€” format directly into qemu_trace!
56        // qemu_trace! is assumed to accept format! style args.
57        qemu_trace!(
58            "[{}] {}: {}\n",
59            record.level(),
60            record.target(),
61            record.args()
62        );
63
64        // Mirror to UEFI console as long as possible.
65        if self.boot_services_available {
66            uefi::println!(
67                "[{}] {}: {}",
68                record.level(),
69                record.target(),
70                record.args()
71            );
72        }
73    }
74
75    fn flush(&self) {
76        // no-op for qemu debug port
77    }
78}
