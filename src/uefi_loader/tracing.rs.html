1//! # Trace output
2
3use crate::TrampolineStackVirtualAddress;
4use kernel_info::boot::KernelBootInfo;
5use kernel_memory_addresses::VirtualAddress;
6
7pub fn trace_boot_info(
8    boot_info: &KernelBootInfo,
9    bi_ptr_va: VirtualAddress,
10    kernel_va: VirtualAddress,
11    trampoline_stack_va: TrampolineStackVirtualAddress,
12) {
13    use log::info;
14
15    info!(
16        concat!(
17            "Boot Info in UEFI Loader:\n",
18            "  Kernel   = {kernel_va:?}\n",
19            "  Trampol. = {trampoline_stack_va:?}\n",
20            "  BI ptr   = {bi_ptr:#018x} (@{bi_mib} MiB)\n",
21            "       VA  = {bi_ptr_va:?}\n",
22            "  MMAP ptr = {mmap_ptr:#018x} (@{mmap_mib} MiB), ",
23            "len = {mmap_len}, desc size = {mmap_desc_size}, ",
24            "desc ver = {mmap_desc_ver}, rsdp addr = {rsdp_addr}\n",
25            "  FB ptr   = {fb_ptr:#018x} (@{fb_mib} MiB), ",
26            "size = {fb_size}, width = {fb_width}, height = {fb_height}, ",
27            "stride = {fb_stride}, format = {fb_fmt}"
28        ),
29        kernel_va = kernel_va,
30        trampoline_stack_va = trampoline_stack_va,
31        bi_ptr = core::ptr::from_ref(boot_info) as usize,
32        bi_mib = (core::ptr::from_ref(boot_info) as usize) / 1024 / 1024,
33        bi_ptr_va = bi_ptr_va,
34        mmap_ptr = boot_info.mmap.mmap_ptr,
35        mmap_mib = boot_info.mmap.mmap_ptr / 1024 / 1024,
36        mmap_len = boot_info.mmap.mmap_len,
37        mmap_desc_size = boot_info.mmap.mmap_desc_size,
38        mmap_desc_ver = usize::try_from(boot_info.mmap.mmap_desc_version).unwrap_or_default(),
39        rsdp_addr = usize::try_from(boot_info.rsdp_addr).unwrap_or_default(),
40        fb_ptr = boot_info.fb.framebuffer_ptr,
41        fb_mib = boot_info.fb.framebuffer_ptr / 1024 / 1024,
42        fb_size = boot_info.fb.framebuffer_size,
43        fb_width = boot_info.fb.framebuffer_width,
44        fb_height = boot_info.fb.framebuffer_height,
45        fb_stride = boot_info.fb.framebuffer_stride,
46        fb_fmt = match boot_info.fb.framebuffer_format {
47            kernel_info::boot::BootPixelFormat::Rgb => "RGB",
48            kernel_info::boot::BootPixelFormat::Bgr => "BGR",
49            kernel_info::boot::BootPixelFormat::Bitmask => "Bitmask",
50            kernel_info::boot::BootPixelFormat::BltOnly => "BltOnly",
51        },
52    );
53}
