1//! # Minimal Bitmap-based Physical Memory Manager (PMM)
2//!
3//! This module provides a minimal, no-heap physical memory manager for 4K frames,
4//! using a bitmap to track free/used frames in a fixed region. It is suitable for
5//! early kernel use or as a foundation for a more advanced PMM.
6//!
7//! ## Features
8//! - Tracks allocation and freeing of 4K frames using a bitmap.
9//! - No heap required; all state is stored inline.
10//! - Can be extended to initialize from a memory map.
11//!
12//! ## Usage Example
13//! ```rust
14//! use kernel_alloc::frame_alloc::BitmapFrameAlloc;
15//! use kernel_vmem::PhysFrameAlloc;
16//! let mut pmm = BitmapFrameAlloc::new();
17//! let frame = pmm.alloc_4k();
18//! if let Some(pa) = frame {
19//!     // Use the physical address...
20//!     pmm.free_4k(pa);
21//! }
22//! ```
23//!
24//! ## Safety
25//! - Only physical addresses within the managed region are tracked.
26//! - The user must ensure that reserved/used frames (e.g., kernel, bootloader) are marked as used before allocation.
27//! - No synchronization is provided; not thread-safe.
28
29use kernel_memory_addresses::{PageSize, PhysicalAddress, PhysicalPage, Size4K};
30use kernel_vmem::PhysFrameAlloc;
31use log::trace;
32
33const PHYS_MEM_START: u64 = 0x0010_0000; // 1 MiB, example
34const PHYS_MEM_SIZE: u64 = 512 * 1024 * 1024; // 512 MiB, example
35const FRAME_SIZE: u64 = Size4K::SIZE;
36const NUM_FRAMES: usize = (PHYS_MEM_SIZE / FRAME_SIZE) as usize;
37
38/// Minimal bitmap-based PMM for 4K frames in a fixed region.
39///
40/// This type manages a fixed region of physical memory, tracking free/used 4K frames
41/// using a bitmap. It supports allocation and freeing, but does not require a heap.
42///
43/// # Example
44/// ```rust
45/// use kernel_alloc::frame_alloc::BitmapFrameAlloc;
46/// use kernel_vmem::PhysFrameAlloc;
47/// let mut pmm = BitmapFrameAlloc::new();
48/// let frame = pmm.alloc_4k();
49/// if let Some(pa) = frame {
50///     // Use the physical address...
51///     pmm.free_4k(pa);
52/// }
53/// ```
54///
55/// # Safety
56/// - Only physical addresses within the managed region are tracked.
57/// - The user must ensure that reserved/used frames (e.g., kernel, bootloader) are marked as used before allocation.
58/// - No synchronization is provided; not thread-safe.
59pub struct BitmapFrameAlloc {
60    bitmap: [u64; NUM_FRAMES.div_ceil(64)],
61    base: u64,
62}
63
64impl Default for BitmapFrameAlloc {
65    fn default() -> Self {
66        Self::new()
67    }
68}
69
70impl BitmapFrameAlloc {
71    #[must_use]
72    pub const fn new() -> Self {
73        Self {
74            bitmap: [0; NUM_FRAMES.div_ceil(64)],
75            base: PHYS_MEM_START,
76        }
77    }
78
79    #[must_use]
80    #[allow(clippy::cast_possible_truncation)]
81    pub const fn manageable_size(&self) -> u64 {
82        (NUM_FRAMES as u64) * FRAME_SIZE
83    }
84
85    /// Mark a frame as used (allocated).
86    pub const fn mark_used(&mut self, frame_idx: usize) {
87        let (word, bit) = (frame_idx / 64, frame_idx % 64);
88        self.bitmap[word] |= 1 << bit;
89    }
90
91    /// Mark a frame as free.
92    pub const fn mark_free(&mut self, frame_idx: usize) {
93        let (word, bit) = (frame_idx / 64, frame_idx % 64);
94        self.bitmap[word] &= !(1 << bit);
95    }
96
97    /// Returns true if the frame is allocated.
98    #[must_use]
99    pub const fn is_used(&self, frame_idx: usize) -> bool {
100        let (word, bit) = (frame_idx / 64, frame_idx % 64);
101        (self.bitmap[word] & (1 << bit)) != 0
102    }
103}
104
105impl PhysFrameAlloc for BitmapFrameAlloc {
106    /// Allocates a single 4 KiB physical frame.
107    ///
108    /// This method iterates over the bitmap looking for the first free bit.
109    /// When it finds one, it:
110    /// 1. Marks the bit as used.
111    /// 2. Computes the corresponding physical address.
112    /// 3. Returns a [`PhysicalPage<Size4K>`] representing that frame.
113    ///
114    /// # Returns
115    /// - `Some(PhysicalPage<Size4K>)` if a free frame was found.
116    /// - `None` if all frames are already allocated.
117    fn alloc_4k(&mut self) -> Option<PhysicalPage<Size4K>> {
118        for (i, word) in self.bitmap.iter_mut().enumerate() {
119            if *word == u64::MAX {
120                continue;
121            }
122
123            for bit in 0..64 {
124                let idx = i * 64 + bit;
125                if idx >= NUM_FRAMES {
126                    break;
127                }
128
129                if (*word & (1 << bit)) != 0 {
130                    continue;
131                }
132
133                *word |= 1 << bit;
134                let pa = self.base + (idx as u64) * FRAME_SIZE;
135                let pa = PhysicalAddress::new(pa);
136                trace!("Allocated 4K frame at {pa}");
137                return Some(PhysicalPage::from_addr(pa));
138            }
139        }
140        None
141    }
142
143    /// Frees a 4 KiB physical frame.
144    ///
145    /// This method clears the corresponding bit in the bitmap,
146    /// marking the frame as free and available for future allocations.
147    ///
148    /// # Arguments
149    /// * `pa` - The physical page to free.
150    ///
151    /// # Safety
152    /// The caller must ensure that:
153    /// - The frame being freed was previously allocated by this allocator.
154    /// - The frame is not in active use.
155    fn free_4k(&mut self, pa: PhysicalPage<Size4K>) {
156        trace!("Freeing 4K frame at {pa}");
157        let idx = ((pa.base().as_u64() - self.base) / FRAME_SIZE) as usize;
158        self.mark_free(idx);
159    }
160}
