3use crate::msr::{Msr, is_canonical_gs_base};
4use crate::{LoadRegisterUnsafe, StoreRegisterUnsafe};
5use bitfield_struct::bitfield;
6use core::ptr::NonNull;

8/// Model-Specific Register: **kernel GS base**.
9///
10/// This MSR holds the *alternate* GS base used by the CPU after executing the
11/// `swapgs` instruction. It allows the kernel to maintain its own GS base
12/// independently of userland.
13///
14/// On `swapgs`, the CPU atomically swaps the contents of
15/// `IA32_GS_BASE` and `IA32_KERNEL_GS_BASE`.
16#[bitfield(u64, order = Lsb)]
17pub struct Ia32KernelGsBaseMsr {
18    #[bits(64)]
19    #[doc(alias = "kernel_gs_base_ptr")]
20    pub ptr: u64,
21}
22
23impl Ia32KernelGsBaseMsr {
24    pub const IA32_KERNEL_GS_BASE: u32 = 0xC000_0102;
25    pub const MSR: Msr = Msr::new(Self::IA32_KERNEL_GS_BASE);
26
27    /// Set the *kernel* GS base that becomes active after `swapgs`.
28    ///
29    /// # Safety
30    /// - CPL0 only; WRMSR at CPL>0 traps.
31    /// - `base` must be a valid, canonical virtual address to kernel per-CPU (or similar).
32    /// - Ensure your `swapgs` usage matches your entry/exit path expectations.
33    #[inline(always)]
34    #[allow(clippy::inline_always)]
35    #[must_use]
36    pub fn with_kernel_gs_base<T>(self, base: NonNull<T>) -> Self {
37        let addr = base.as_ptr() as u64;
38        debug_assert!(
39            is_canonical_gs_base(addr),
40            "non-canonical KERNEL_GS base: {addr:#x}"
41        );
42        self.with_ptr(addr)
43    }
44}
45
46#[cfg(feature = "asm")]
47impl LoadRegisterUnsafe for Ia32KernelGsBaseMsr {
48    #[inline(always)]
49    #[allow(clippy::inline_always)]
50    unsafe fn load_unsafe() -> Self {
51        let msr = unsafe { Self::MSR.load_raw() };
52        Self::from_bits(msr)
53    }
54}
55
56#[cfg(feature = "asm")]
57impl StoreRegisterUnsafe for Ia32KernelGsBaseMsr {
58    #[inline(always)]
59    #[allow(clippy::inline_always)]
60    unsafe fn store_unsafe(self) {
61        unsafe { Self::MSR.store_raw(self.into_bits()) }
62    }
