5/// `IA32_STAR` — System Call Target & Segment Selectors (MSR `0xC000_0081`).
6///
7/// In 64-bit mode:
8///
9/// - `syscall` uses `IA32_STAR[47:32]` to derive kernel CS/SS.
10/// - `sysret` uses `IA32_STAR[63:48]` to derive user CS/SS.
11/// - `IA32_LSTAR` provides the 64-bit RIP target for `syscall`.
12///
13/// In compatibility mode, `IA32_STAR[31:0]` holds the 32-bit EIP target for `syscall`.
14#[bitfield(u64)]
15pub struct Ia32Star {
16    /// Bits 0–31 — Compatibility-mode `syscall` EIP.
17    ///
18    /// Used only when executing `syscall` from IA-32e compatibility mode.
19    /// Ignored for 64-bit `syscall`, which uses `IA32_LSTAR`.
20    #[bits(32, access = RO)]
21    pub compat_syscall_eip: u32,
22
23    /// Bits 32–47 — Kernel code segment selector base for `syscall`.
24    ///
25    /// On `syscall` in 64-bit mode:
26    /// ```text
27    ///   CS ← (this & 0xFFFC)
28    ///   SS ← (this + 8)
29    /// ```
30    ///
31    /// You typically store your kernel CS selector here (e.g. `KERNEL_CS_SEL`, `0x08`).
32    #[bits(16)]
33    pub syscall_cs_selector: u16,
34
35    /// Bits 48–63 — User code segment selector base for `sysret`.
36    ///
37    /// On `sysret` in 64-bit mode:
38    /// ```text
39    ///   CS ← (this + 16) | 3
40    ///   SS ← (this +  8) | 3
41    /// ```
42    ///
43    /// You typically store your user CS selector here (e.g. `USER_CS_SEL`, `0x1b`).
44    #[bits(16)]
45    pub sysret_cs_selector: u16,
46}
47
48impl Ia32Star {
49    /// MSR index for `IA32_STAR`.
50    pub const IA32_STAR: u32 = 0xC000_0081;
51
52    /// The MSR.
53    pub const MSR: Msr = Msr::new(Self::IA32_STAR);
54
55    /// Helper to build a STAR value for a pure 64-bit kernel.
56    ///
57    /// `kernel_cs` and `user_cs` are the *selectors* (e.g. `0x08` and `0x1b`).
58    #[must_use]
59    pub fn new_64bit_raw(kernel_cs: u16, user_cs: u16) -> Self {
60        // Assumption:
61        // kcode -> kdata -> udata -> ucode
62
63        // Small helpers for raw selectors.
64        #[inline]
65        const fn gdt_index(sel: u16) -> u16 {
66            sel >> 3
67        }
68
69        #[inline]
70        const fn rpl(sel: u16) -> u16 {
71            sel & 0b11
72        }
73
74        // Extract and validate GDT indices
75        let kidx = gdt_index(kernel_cs);
76        let uidx = gdt_index(user_cs);
77
78        // SYSRET requires:
79        //   user_ss_index = user_cs_index - 1
80        // because SS uses (base + 8), CS uses (base + 16).
81        debug_assert_ne!(kidx, 0);
82        debug_assert_eq!(rpl(kernel_cs), 0, "kernel CS must be Ring0");
83        debug_assert_ne!(uidx, 0, "User CS selector at GDT index 0 is invalid");
84
85        let user_ss_index = uidx - 1;
86
87        // Compute STAR base value for SYSRET
88        //
89        // STAR[63:48] = raw 16-bit base selector WITHOUT RPL bits.
90        //
91        //   SS = (base + 8 ) | 3  → must equal (user_ss_index << 3) | 3
92        //   CS = (base + 16) | 3  → must equal (user_cs_index << 3) | 3
93        //
94        // Solve:
95        //   base = (user_ss_index << 3) - 8
96        //
97        let base_no_rpl: u16 = (user_ss_index << 3).wrapping_sub(8);
98
99        Self::new()
100            // Hardware ignores the RPL bits for the syscall CS selector in STAR,
101            // but storing the full selector is fine; only bits 15:3 matter.
102            .with_syscall_cs_selector(kernel_cs)
103            .with_sysret_cs_selector(base_no_rpl)
104    }
105}
