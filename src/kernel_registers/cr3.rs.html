5/// CR3 — Page-Map Level-4 Base Register (IA-32e, PCID disabled).
6///
7/// Holds the physical base address of the PML4 table and cache-control flags
8/// for PML4 walks. Assumes standard 4 KiB alignment and no PCID (CR4.PCIDE = 0).
9#[bitfield(u64)]
10pub struct Cr3 {
11    /// Bits 0–2 — Reserved (must be 0).
12    #[bits(3)]
13    pub reserved0: u8,
14
15    /// Bit 3 — PWT: Page-level Write-Through for PML4.
16    ///
17    /// Controls write-through vs write-back caching when accessing the PML4
18    /// via CR3.
19    pub pwt: bool,
20
21    /// Bit 4 — PCD: Page-level Cache Disable for PML4.
22    ///
23    /// When set, disables caching for PML4 accesses.
24    pub pcd: bool,
25
26    /// Bits 5–11 — Reserved (must be 0 when written).
27    #[bits(7)]
28    pub reserved1: u8,
29
30    /// Bits 12–51 — PML4 physical base >> 12.
31    ///
32    /// These bits store the physical base address of the PML4 table, shifted
33    /// right by 12 (4 KiB alignment). To get the full physical address:
34    /// `pml4_base_phys = pml4_base_4k << 12`.
35    #[bits(40)]
36    pml4_base_4k: u64,
37
38    /// Bits 52–63 — Reserved.
39    #[bits(12)]
40    pub reserved2: u16,
41}
42
43impl Cr3 {
44    /// Create a `Cr3` value from a PML4 physical base address and flags.
45    ///
46    /// `pml4_phys` must be 4 KiB-aligned.
47    #[must_use]
48    pub fn from_pml4_phys(pml4_phys: PhysicalAddress, pwt: bool, pcd: bool) -> Self {
49        debug_assert_eq!(
50            pml4_phys.as_u64() & 0xFFF,
51            0,
52            "PML4 base must be 4K-aligned"
53        );
54        let mut cr3 = Self::new();
55        cr3.set_pwt(pwt);
56        cr3.set_pcd(pcd);
57        cr3.set_pml4_base_4k(pml4_phys.as_u64() >> 12);
58        cr3
59    }
60
61    /// Return the full physical address of the PML4 base.
62    #[must_use]
63    pub fn pml4_phys(&self) -> PhysicalAddress {
64        // In 4- and 5-level paging, CR3[51:12] is the base. Upper bits should be zero.
65        let bits = self.into_bits();
66        debug_assert_eq!(bits >> 52, 0, "CR3 has nonzero high bits: {bits:#018x}");
67
68        PhysicalAddress::new(self.pml4_base_4k() << 12)
69    }
70}

72#[cfg(feature = "asm")]
73impl LoadRegisterUnsafe for Cr3 {
74    unsafe fn load_unsafe() -> Self {
75        let mut cr3: u64;
76        unsafe {
77            core::arch::asm!("mov {}, cr3", out(reg) cr3, options(nomem, nostack, preserves_flags));
78        }
79        Self::from_bits(cr3)
80    }
81}
