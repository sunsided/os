1use bitfield_struct::bitfield;

3/// Architectural RFLAGS model for x86-64.
4///
5/// Bits that are architecturally fixed in 64-bit mode are modeled as
6/// `#[bits(default=.., access=RO)]` so they cannot be changed.
7///
8/// Bits that may legally be manipulated (and masked via `IA32_FMASK`)
9/// are exposed as normal read/write bools or small enums.
10#[bitfield(u64, order = Lsb)]
11pub struct Rflags {
12    /// Carry Flag
13    pub cf_carry: bool, // 0
14
15    /// Always 1 in 64-bit mode.
16    #[bits(default = true)]
17    _always1: bool, // 1
18
19    /// Parity Flag
20    pub pf_parity: bool, // 2
21
22    /// Reserved (always 0)
23    #[bits(default = false)]
24    _rsvd3: bool, // 3
25
26    /// Adjust Flag
27    pub af_adjust: bool, // 4
28
29    /// Reserved (always 0)
30    #[bits(default = false)]
31    _rsvd5: bool, // 5
32
33    /// Zero Flag
34    pub zf_zero: bool, // 6
35
36    /// Sign Flag
37    pub sf_sign: bool, // 7
38
39    /// Trap Flag
40    pub tf_trap: bool, // 8
41
42    /// Interrupt Enable Flag
43    pub if_interrupt_enable: bool, // 9
44
45    /// Direction Flag
46    pub df_direction: bool, // 10
47
48    /// Overflow Flag
49    pub of_overflow: bool, // 11
50
51    /// I/O Privilege Level (2 bits)
52    #[bits(2)]
53    pub iopl: u8, // 12–13
54
55    /// Nested Task
56    pub nt_nested: bool, // 14
57
58    /// Reserved (always 0 in x86-64)
59    #[bits(default = false)]
60    _rsvd15: bool, // 15
61
62    /// Resume Flag
63    pub rf_resume: bool, // 16
64
65    /// Virtual 8086 mode — must be 0 in 64-bit mode.
66    #[bits(default = false)]
67    _vm: bool, // 17
68
69    /// Alignment Check
70    pub ac_alignment_check: bool, // 18
71
72    /// Virtual Interrupt Flag
73    pub vif_virtual_interrupt: bool, // 19
74
75    /// Virtual Interrupt Pending
76    pub vip_virtual_interrupt_pending: bool, // 20
77
78    /// ID Flag: allows toggling CPUID.
79    pub id_cpuid: bool, // 21
80
81    /// Reserved 22–63 (all zero)
82    #[bits(42, default = false)]
83    _reserved_rest: u64,
84}
