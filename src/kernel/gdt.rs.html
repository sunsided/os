1//! # Global Descriptor Table (GDT) & Task State Segment (TSS) descriptor wiring for long mode
2//!
3//! In 64-bit mode (“long mode”), classic segmentation for code/data is largely
4//! disabled, but **segment selectors still matter**:
5//!
6//! - They distinguish **code vs. data/stack** segments.
7//! - They carry the **Descriptor Privilege Level (DPL)** used by the CPU to
8//!   enforce privilege transitions (Ring-0 ↔ Ring-3).
9//! - They locate the **Task State Segment (TSS)** through a **16-byte system
10//!   descriptor** in the GDT so the CPU can fetch `rsp0` (kernel stack pointer)
11//!   and IST stacks on privilege changes and fault handling.
12//!
13//! ## Why you need this in a 64-bit kernel
14//! - Returning to userland with `iretq` requires **user code/data descriptors**
15//!   in the GDT and using their **selectors** (with `RPL=3`) in the `iret` frame.
16//! - Reliable user→kernel transitions require a loaded **TSS** (via `ltr`),
17//!   because the CPU uses `tss.rsp0` as the **Ring-0 stack** on entry.
18//!
19//! ## GDT layout used here
20//! Index | Selector | Meaning
21//! ------|----------|--------
22//! 0     | 0x00     | Null
23//! 1     | 0x08     | Kernel code (64-bit, DPL=0; [`KERNEL_CS_SEL`])
24//! 2     | 0x10     | Kernel data (DPL=0; [`KERNEL_DS_SEL`]
25//! 3     | 0x18     | User   code (64-bit, DPL=3) → with RPL=3: **0x1b** ([`USER_CS_SEL`])
26//! 4     | 0x20     | User   data (DPL=3)         → with RPL=3: **0x23** ([`USER_DS_SEL`])
27//! 5/6   | 0x28     | TSS (16-byte system descriptor; [`TSS_SYS_SEL`])
28//!
29//! This module builds a typed GDT, loads it with `lgdt`, refreshes the data
30//! segments, and loads the Task Register (`ltr`) with the TSS selector.
31//!
32//! ## SMP / per-CPU note
33//! The **TSS is per-CPU**. If you run multiple CPUs, allocate a TSS (and
34//! typically a GDT) per CPU and call the init routine on each CPU during bring-up.
35//!
36//! ## Preconditions
37//! - Long mode and paging are enabled.
38//! - The GDT (and TSS) memory is mapped and **writable during setup** and
39//!   **readable thereafter** at the virtual addresses you pass in.
40//! - Interrupts should be masked during the switch to avoid using half-set state.
41

82// Well-known selectors matching the GDT layout in `gdt.rs`.
83//
84// The `*_SEL` are typed wrappers; the `*_` constants are the encoded `u16`
85// values (useful for inline asm or iret frames).
86pub const KERNEL_CS_SEL: SegmentSelector<CodeSel> = SegmentSelector::<CodeSel>::new(1, Rpl::Ring0);
87pub const KERNEL_DS_SEL: SegmentSelector<DataSel> = SegmentSelector::<DataSel>::new(2, Rpl::Ring0);
88pub const USER_DS_SEL: SegmentSelector<DataSel> = SegmentSelector::<DataSel>::new(3, Rpl::Ring3);
89pub const USER_CS_SEL: SegmentSelector<CodeSel> = SegmentSelector::<CodeSel>::new(4, Rpl::Ring3);
90pub const TSS_SYS_SEL: SegmentSelector<TssSel> = SegmentSelector::<TssSel>::new(5);
91
92// Encoded selector numbers as `u16` (what the CPU actually loads).
93pub const KERNEL_CS: u16 = KERNEL_CS_SEL.encode(); // 0x08
94pub const KERNEL_DS: u16 = KERNEL_DS_SEL.encode(); // 0x10
95pub const USER_CS: u16 = USER_CS_SEL.encode(); // 0x1b
96pub const USER_DS: u16 = USER_DS_SEL.encode(); // 0x23
97pub const TSS_SEL: u16 = TSS_SYS_SEL.encode(); // 0x28
98
99// Compile-time sanity checks for selectors and descriptor sizes.
100#[allow(clippy::items_after_statements)]
101const _: () = {
102    // Expected raw selector numbers (given the GDT layout in this file).
103    assert!(KERNEL_CS == 0x08);
104    assert!(KERNEL_DS == 0x10);
105    assert!(USER_DS == 0x1b);
106    assert!(USER_CS == 0x23);
107    assert!(TSS_SEL == 0x28);
108
109    // Encoding formula: (index << 3) | (TI=0) | RPL
110    const fn enc(index: u16, rpl: u16) -> u16 {
111        (index << 3) | rpl
112    }
113
114    assert!(KERNEL_CS == enc(1, 0)); // kernel code: index=1, RPL=0
115    assert!(KERNEL_DS == enc(2, 0)); // kernel data: index=2, RPL=0
116    assert!(USER_DS == enc(3, 3)); // user   data: index=4, RPL=3
117    assert!(USER_CS == enc(4, 3)); // user   code: index=3, RPL=3
118    assert!(TSS_SEL == enc(5, 0)); // TSS (low): index=5, RPL=0
119
120    // Typed selectors must produce the same raw values.
121    assert!(KERNEL_CS == KERNEL_CS_SEL.encode());
122    assert!(KERNEL_DS == KERNEL_DS_SEL.encode());
123    assert!(USER_CS == USER_CS_SEL.encode());
124    assert!(USER_DS == USER_DS_SEL.encode());
125    assert!(TSS_SEL == TSS_SYS_SEL.encode());
126};

128/// Virtual address used in descriptor-table pointers (with paging on).
129///
130/// In long mode `lgdt` expects a **linear (virtual) address** when paging is enabled.
131pub type LinearAddress = VirtualAddress;
132
133/// Pointer format required by `lgdt`.
134///
135/// The CPU reads exactly `limit+1` bytes starting at `base` to load the GDT.
136#[repr(C, packed)]
137struct DescTablePtr {
138    /// Size of the table **minus one** in bytes.
139    limit: u16,
140    /// Base **linear (virtual) address** of the table in memory.
141    base: LinearAddress,
142}
143
144/// The complete GDT for the bootstrap CPU.
145///
146/// Layout matches the table described in this module-level doc. The TSS occupies
147/// two consecutive entries (a 16-byte system descriptor).
148#[repr(C, align(16))]
149pub struct Gdt {
150    /// Null descriptor (must be present at index 0).
151    null: Desc64, // 0
152    /// Kernel code segment (64-bit, DPL=0).
153    kcode: Desc64, // 1
154    /// Kernel data/stack segment (DPL=0).
155    /// Must be one index after `kcode` for `SYSCALL`.
156    kdata: Desc64, // 2
157    /// User data/stack segment (DPL=3).
158    /// Must be one index before `ucode` for `SYSRET`.
159    udata: Desc64, // 3
160    /// User code segment (64-bit, DPL=3).
161    ucode: Desc64, // 4
162    /// 64-bit Available TSS descriptor (low+high).
163    tss: TssDesc64, // 5 & 6 (16-byte system descriptor)
164}
165
166impl Default for Gdt {
167    fn default() -> Self {
168        Self::new()
169    }
170}
171
172impl Gdt {
173    pub const fn new_with_tss(tss: TssDesc64) -> Self {
174        Self {
175            null: Desc64 { raw: 0 },
176            kcode: Desc64::from_code_dpl(Dpl::Ring0), // kernel code: DPL=0
177            kdata: Desc64::from_data_dpl(Dpl::Ring0), // kernel data: DPL=0
178            udata: Desc64::from_data_dpl(Dpl::Ring3), // user   data: DPL=3
179            ucode: Desc64::from_code_dpl(Dpl::Ring3), // user   code: DPL=3
180            tss,
181        }
182    }
183
184    pub const fn new() -> Self {
185        Self::new_with_tss(TssDesc64::new(VirtualAddress::zero(), 0))
186    }
187}

189/// Load a GDT with `lgdt`.
190///
191/// # Safety
192/// - `gdt` must point to a valid, fully initialized table whose memory will
193///   remain **mapped and readable** for the lifetime of the CPU.
194/// - Callers must ensure no interrupts or faults observe a half-installed state.
195#[inline]
196#[allow(clippy::cast_possible_truncation)]
197unsafe fn load_gdt(gdt: &Gdt) {
198    let ptr = DescTablePtr {
199        limit: (size_of::<Gdt>() - 1) as u16,
200        base: LinearAddress::from_ptr(&raw const *gdt),
201    };
202
203    unsafe {
204        core::arch::asm!(
205            "lgdt [{}]",
206            in(reg) &raw const ptr,
207            options(readonly, nostack, preserves_flags)
208        );
209    }
210}

212/// Load the Task Register with a TSS selector.
213///
214/// The selector must refer to a **present 64-bit Available TSS** system descriptor
215/// in the current GDT.
216///
217/// # Safety
218/// - The GDT must be active and contain a valid TSS descriptor at `sel`.
219/// - The TSS memory must remain **resident**; the CPU reads from it on traps and
220///   privilege changes.
221#[inline]
222unsafe fn load_task_register(sel: SegmentSelector<TssSel>) {
223    let sel = sel.encode();
224    unsafe {
225        core::arch::asm!(
226            "ltr {0:x}",
227            in(reg) sel,
228            options(nostack, preserves_flags)
229        );
230    }
231}

233/// Initialize and load **GDT + TSS** for the bootstrap CPU.
234///
235/// - Programs the TSS with `rsp0` (kernel entry stack) and optional `IST1`.
236/// - Builds a GDT with kernel/user code+data descriptors and a 64-bit TSS descriptor.
237/// - Executes `lgdt`, refreshes data segments (DS/ES/SS), and executes `ltr`.
238///
239/// Call exactly **once per CPU** (with per-CPU TSS/GDT if SMP).
240///
241/// ### Parameters
242/// - `kernel_stack_top`: top of the Ring-0 stack (used on CPL change to 0).
243/// - `ist1_top`: optional top of an **IST1** stack for critical handlers.
244///
245/// ### Safety / Ordering
246/// - Run with interrupts disabled.
247/// - Long mode + paging are already enabled.
248/// - TSS storage and GDT memory must remain mapped and accessible thereafter.
249///
250/// ### Example
251/// ```ignore
252/// // During boot on BSP:
253/// init_gdt_and_tss(kernel_stack_top, Some(double_fault_stack_top));
254/// ```
255#[allow(clippy::cast_possible_truncation)]
256pub fn init_gdt_and_tss(
257    p: &mut PerCpu,
258    kernel_stack_top: VirtualAddress,
259    ist1_top: VirtualAddress,
260) {
261    // Initialize TSS contents first.
262    init_tss(p, kernel_stack_top, ist1_top);
263    let tss_base = p.tss_base();
264    let tss_limit = (size_of::<Tss64>() - 1) as u32;
265
266    // Build GDT with typed descriptors (no raw bit twiddling here).
267    p.gdt = Gdt::new_with_tss(TssDesc64::new(tss_base, tss_limit));
268
269    // Load GDT + TR and refresh data segments.
270    #[allow(static_mut_refs)]
271    unsafe {
272        // Load the GDTR with this CPU's GDT
273        load_gdt(&p.gdt);
274
275        // Refresh data segments to kernel data
276        let kdata_sel = p.selectors.kernel_ds.encode();
277        core::arch::asm!(
278            "mov ds, {0:x}",
279            "mov es, {0:x}",
280            "mov ss, {0:x}",
281            in(reg) kdata_sel,
282            options(nostack, preserves_flags)
283        );
284
285        // Far reload of CS = 0x08 (kernel code). Use retfq trick in long mode.
286        let kcs: u16 = p.selectors.kernel_cs.encode(); // 0x08
287        core::arch::asm!(
288            // push target CS and RIP, then far return
289            "push {cs}",
290            "lea rax, [rip + 2f]",
291            "push rax",
292            "retfq",
293            "2:",
294            cs = in(reg) u64::from(kcs),
295            out("rax") _,
296            options(nostack)
297        );
298
299        // Load TR with the TSS selector (0x28).
300        load_task_register(p.selectors.tss);
301    }
