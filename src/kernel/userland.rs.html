12pub unsafe fn enter_user_mode(entry: VirtualAddress, user_sp: VirtualAddress) -> ! {
13    let rip = entry.as_u64();
14    let cs = u64::from(USER_CS) | 3;
15    let ss = u64::from(USER_DS) | 3;
16    let rsp = user_sp.as_u64();
17    let rflags: u64 = 0x202;
18
19    // prove we got here
20    info!("Entering user mode ...");
21
22    unsafe {
23        core::arch::asm!(
24            "push {ss}",
25            "push {rsp}",
26            "push {rflags}",
27            "push {cs}",
28            "push {rip}",
29            "iretq",
30            ss = in(reg) ss, rsp = in(reg) rsp, rflags = in(reg) rflags,
31            cs = in(reg) cs, rip = in(reg) rip,
32            options(noreturn)
33        )
34    }
35}

37pub fn boot_single_user_task(vmm: &mut KernelVmm) -> ! {
38    // Choose stable user VAs
39    let code_va = VirtualAddress::new(0x0000_4000_0000);
40    let ustack_top = VirtualAddress::new(0x0000_7fff_f000);
41
42    let blob = userland_demo::user_demo_bytes();
43
44    info!("Mapping user demo ...");
45    let (entry, user_sp_top) =
46        map_user_demo(vmm, code_va, ustack_top, blob).expect("map user demo");
47
48    info!("About to enter user mode ...");
49    log_ctrl_bits();
50
51    // TODO: Remove later
52    alloc::debug::dump_walk(&HhdmPhysMapper, VirtualAddress::new(0x0000_0000_4000_0000));
53
54    info!("About to flush TLB ...");
55    unsafe {
56        vmm.local_tlb_flush_all();
57    }
58
59    unsafe { enter_user_mode(entry, user_sp_top) } // iretq; never returns
60}

62#[allow(clippy::similar_names)]
63fn map_user_demo<M: PhysMapper, A: PhysFrameAlloc>(
64    vmm: &mut Vmm<'_, M, A>,
65    code_va: VirtualAddress,
66    ustack_top: VirtualAddress,
67    blob: &[u8],
68) -> Result<(VirtualAddress, VirtualAddress), VmmError> {
69    let _guard = SmapGuard::enter();
70
71    // non-leaf: allow user traversal (U/S=1), WB, Present, Write=1 (harmless), NX don't-care
72    let nonleaf = VirtualMemoryPageBits::user_table_wb_exec(); // must set US=1
73
74    // leaf for CODE (RX): Present=1, User=1, Write=0, NX=0
75    let leaf_rx = VirtualMemoryPageBits::user_leaf_code_wb(); // NX=0, US=1
76
77    // leaf for DATA/STACK (RW, NX): Present=1, User=1, Write=1, NX=1
78    let leaf_rw = VirtualMemoryPageBits::user_leaf_data_wb(); // NX=1, US=1
79
80    // Code: map RW temporarily, copy, then flip to RX
81    let code_len = blob.len() as u64;
82    let code_len_4k = (code_len + Size4K::SIZE - 1) & !(Size4K::SIZE - 1);
83
84    vmm.map_anon_4k_pages(
85        AllocationTarget::User,
86        code_va,
87        0,           // no guard for code
88        code_len_4k, // whole pages
89        nonleaf,
90        leaf_rw, // writable to copy in
91    )?;
92
93    unsafe {
94        vmm.copy_to_mapped_user(code_va, blob)?;
95    }
96
97    // Now switch code pages to RX
98    vmm.make_region_rx(code_va, code_len, nonleaf, leaf_rx)?;
99
100    // Stack: 32 KiB RW with a 4 KiB guard below
101    let stack_size = 8 * Size4K::SIZE;
102    let guard = Size4K::SIZE;
103    let stack_base = VirtualAddress::new(ustack_top.as_u64() - guard - stack_size);
104
105    vmm.map_anon_4k_pages(
106        AllocationTarget::User,
107        stack_base,
108        guard, // leave guard unmapped (fault on underflow)
109        stack_size,
110        nonleaf,
111        leaf_rw, // writable, NX
112    )?;
113
114    Ok((code_va, ustack_top))
115}
