29            let r = cpuid(LEAF_15H, 0);

41        }
42    }

44    /// Compute TSC frequency if enough info is present.
45    pub fn tsc_hz(&self) -> Option<u64> {

47            Some(u64::from(self.crystal_hz) * u64::from(self.numer) / u64::from(self.denom))
48        } else {
