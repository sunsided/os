1use crate::cpuid::cpuid;
2
3const LEAF_00H: u32 = 0b00;
4const LEAF_MAX_EXTENDED: u32 = 0x8000_0000;
5
6#[derive(Copy, Clone)]
7pub struct CpuidRanges {
8    pub max_basic: u32,
9    pub max_extended: u32,
10    pub vendor: CpuVendor,
11}
12
13#[derive(Copy, Clone, Debug, Eq, PartialEq)]
14pub enum CpuVendor {
15    Intel,
16    Amd,
17    Other,
18}
19
20impl CpuidRanges {
21    pub unsafe fn read() -> Self {
22        let b0 = unsafe { cpuid(LEAF_00H, 0) };
23        let max_basic = b0.eax;
24
25        let v = [b0.ebx, b0.edx, b0.ecx]; // e.g., "GenuineIntel", "AuthenticAMD"
26        let ptr: &[u8] = unsafe { core::slice::from_raw_parts(v.as_ptr().cast::<u8>(), 12) };
27
28        let vendor = match unsafe { core::str::from_utf8_unchecked(ptr).trim_end_matches('\0') } {
29            "GenuineIntel" => CpuVendor::Intel,
30            "AuthenticAMD" => CpuVendor::Amd,
31            _ => CpuVendor::Other,
32        };
33
34        let e0 = unsafe { cpuid(LEAF_MAX_EXTENDED, 0) };
35        let max_extended = e0.eax;
36
37        Self {
38            max_basic,
39            max_extended,
40            vendor,
41        }
42    }
43
44    #[inline]
45    pub const fn has_basic(&self, leaf: u32) -> bool {
46        leaf <= self.max_basic
47    }
48
49    #[inline]
50    pub const fn has_ext(&self, leaf: u32) -> bool {
51        leaf >= 0x8000_0000 && leaf <= self.max_extended
52    }
53}
54
55impl CpuVendor {
56    pub const fn as_str(self) -> &'static str {
57        match self {
58            Self::Intel => "Intel",
59            Self::Amd => "AMD",
60            Self::Other => "Other",
61        }
62    }
63}
