6/// CPUID.01H — Feature Information (a.k.a. “leaf 1”).
7///
8/// Returns processor version info (EAX), brand/CLFLUSH/logical count/APIC ID (EBX),
9/// and the classic feature flags (ECX/EDX). This wrapper parses the common
10/// identification fields and exposes ECX/EDX via your bitfield views.
11///
12/// Reference: Intel SDM Vol. 2A, “CPUID—CPU Identification”, leaf 01H.
13#[derive(Copy, Clone, Debug)]
14pub struct Leaf01h {
15    pub eax: Leaf1Eax,
16    pub ebx: Leaf1Ebx,
17    pub ecx: Leaf1Ecx,
18    pub edx: Leaf1Edx,
19}
20
21impl Leaf01h {
22    /// # Safety
23    /// The caller must ensure that the `cpuid` instruction is available and leaf `0x01` exists.
24    pub unsafe fn new() -> Self {
25        unsafe {
26            let r = cpuid(LEAF_01H, 0);
27            Self::from(r)
28        }
29    }
30
31    /// Query CPUID.01H if supported; returns `None` if `ranges` says leaf 1 is absent.
32    #[inline]
33    pub unsafe fn read(ranges: &CpuidRanges) -> Option<Self> {
34        if !ranges.has_basic(LEAF_01H) {
35            return None;
36        }
37
38        unsafe {
39            let r = cpuid(LEAF_01H, 0);
40            Some(Self::from(r))
41        }
42    }
43
44    /// # Safety
45    /// The caller must ensure that the passed [`CpuidResult`] belongs to a valid leaf `0x01` entry.
46    pub const unsafe fn from(r: CpuidResult) -> Self {
47        Self {
48            eax: Leaf1Eax::from_bits(r.eax),
49            ebx: Leaf1Ebx::from_bits(r.ebx),
50            ecx: Leaf1Ecx::from_bits(r.ecx),
51            edx: Leaf1Edx::from_bits(r.edx),
52        }
53    }
54
55    #[inline]
56    pub const fn has_x2apic(&self) -> bool {
57        self.ecx.x2apic()
58    }
59
60    #[inline]
61    pub const fn avx_usable(&self) -> bool {
62        self.ecx.avx() && self.ecx.xsave() && self.ecx.osxsave()
63    }
64
65    #[inline]
66    pub const fn initial_apic_id(&self) -> u8 {
67        self.ebx.initial_apic_id()
68    }
69
70    #[inline]
71    pub const fn logical_cpus_legacy(&self) -> u8 {
72        self.ebx.logical_processor_count()
73    }
74
75    #[inline]
76    pub fn family(&self) -> u16 {
77        self.eax.effective_family()
78    }
79
80    #[inline]
81    pub const fn model(&self) -> u8 {
82        self.eax.effective_model()
83    }
84
85    #[inline]
86    pub const fn stepping(&self) -> u8 {
87        self.eax.stepping()
88    }
89}

91/// CPUID.01H:EAX — Version Information.
92///
93/// Raw fields as defined by the SDM; helpers compute the *effective* model/family.
94/// Reference: Intel SDM Vol. 2A, CPUID leaf 01H, EAX layout.
95#[bitfield(u32)]
96pub struct Leaf1Eax {
97    /// Stepping ID (bits 3:0).
98    #[bits(4)]
99    stepping: u8,
100    /// Base model (bits 7:4).
101    #[bits(4)]
102    model: u8,
103    /// Base family (bits 11:8).
104    #[bits(4)]
105    family: u8,
106    /// Processor type (bits 13:12).
107    #[bits(2)]
108    cpu_type: u8,
109    /// Reserved (bits 15:14).
110    #[bits(2)]
111    _rsv14_15: u8,
112    /// Extended model (bits 19:16).
113    #[bits(4)]
114    ext_model: u8,
115    /// Extended family (bits 27:20).
116    #[bits(8)]
117    ext_family: u16,
118    /// Reserved (bits 31:28).
119    #[bits(4)]
120    _rsv28_31: u8,
121}
122
123impl Leaf1Eax {
124    /// Effective family per SDM:
125    /// if base family == 0x0F → base + `ext_family`, else base.
126    #[inline]
127    pub fn effective_family(self) -> u16 {
128        let fam = u16::from(self.family());
129        if fam == 0x0F {
130            fam + self.ext_family()
131        } else {
132            fam
133        }
134    }
135
136    /// Effective model per SDM:
137    /// if base family in {0x06, 0x0F} → `base_model` | (`ext_model` << 4), else `base_model`.
138    #[inline]
139    pub const fn effective_model(self) -> u8 {
140        let fam = self.family();
141        let base = self.model();
142        if fam == 0x06 || fam == 0x0F {
143            base | (self.ext_model() << 4)
144        } else {
145            base
146        }
147    }
148}
