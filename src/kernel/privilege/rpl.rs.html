1//! Requested Privilege Level (RPL).
2//!
3//! RPL lives in the **low 2 bits of a segment selector** supplied by the requester
4//! (the code performing the load). Don’t confuse it with:
5//! - **CPL** (Current Privilege Level): taken from the running `CS`.
6//! - **DPL** (Descriptor Privilege Level): stored in the target descriptor (GDT/LDT/gate).
7//!
8//! For *data* segment loads, the CPU checks: `max(CPL, RPL) ≤ DPL`.
9//! For `SS` loads in long mode: `CPL == RPL == DPL`.
10//!
11//! This module provides the `Rpl` enum plus helpers to extract/apply it to
12//! a selector, and convenience checks that mirror the architecture’s rules.
13
14#![allow(dead_code)]
15
16use crate::privilege::Ring;
17
18/// RPL mask in a 16-bit selector.
19pub const RPL_MASK: u16 = 0b11;
20
21#[derive(Copy, Clone, Eq, PartialEq, Debug)]
22#[repr(u8)]
23pub enum Rpl {
24    Ring0 = 0,
25    #[allow(deprecated)]
26    Ring1 = 1,
27    #[allow(deprecated)]
28    Ring2 = 2,
29    Ring3 = 3,
30}
31
32impl Rpl {
33    /// Construct from a `Ring` value.
34    #[inline]
35    #[must_use]
36    pub const fn from_ring(ring: Ring) -> Self {
37        match ring {
38            Ring::Ring0 => Self::Ring0,
39            #[allow(deprecated)]
40            Ring::Ring1 => Self::Ring1,
41            #[allow(deprecated)]
42            Ring::Ring2 => Self::Ring2,
43            Ring::Ring3 => Self::Ring3,
44        }
45    }
46
47    /// Convert back to a `Ring`.
48    #[inline]
49    #[must_use]
50    pub const fn to_ring(self) -> Ring {
51        match self {
52            Self::Ring0 => Ring::Ring0,
53            #[allow(deprecated)]
54            Self::Ring1 => Ring::Ring1,
55            #[allow(deprecated)]
56            Self::Ring2 => Ring::Ring2,
57            Self::Ring3 => Ring::Ring3,
58        }
59    }
60
61    /// Encode as the low two bits of a selector.
62    #[inline]
63    #[must_use]
64    pub const fn into_bits(self) -> u16 {
65        self as u16
66    }
67
68    /// Decode from the low two bits.
69    #[inline]
70    #[must_use]
71    pub const fn from_bits(value_low2: u16) -> Self {
72        match value_low2 & RPL_MASK {
73            0 => Self::Ring0,
74            1 => Self::Ring1,
75            2 => Self::Ring2,
76            _ => Self::Ring3,
77        }
78    }
79
80    /// Extract `RPL` from a 16-bit segment selector value.
81    #[inline]
82    #[must_use]
83    pub const fn from_selector(selector: u16) -> Self {
84        Self::from_bits(selector & RPL_MASK)
85    }
86
87    /// Return `selector` with its `RPL` bits replaced by `self`.
88    #[inline]
89    #[must_use]
90    pub const fn apply_to_selector(self, selector: u16) -> u16 {
91        (selector & !RPL_MASK) | self.into_bits()
92    }
93
94    /// Build a selector from `{index, ti}` and this `RPL`.
95    ///
96    /// - `index`: descriptor index in the table (GDT/LDT)
97    /// - `ti`: table indicator (0 = GDT, 1 = LDT)
98    ///
99    /// Layout: `selector = (index << 3) | (ti << 2) | rpl`.
100    #[inline]
101    #[must_use]
102    pub const fn build_selector(self, index: u16, ti: bool) -> u16 {
103        (index << 3) | ((ti as u16) << 2) | self.into_bits()
104    }
105
106    /// The *effective requester level* that participates in data-segment checks:
107    /// `max(CPL, RPL)`.
108    #[inline]
109    #[must_use]
110    pub const fn effective_with(self, cpl: Ring) -> Ring {
111        let c = cpl as u8;
112        let r = self as u8;
113        if c >= r { cpl } else { self.to_ring() }
114    }
115
116    /// CPU-like predicate for loading a *data* segment against a given `DPL`.
117    /// Returns `true` iff `max(CPL, RPL) ≤ DPL`.
118    ///
119    /// (Keep this here for tests/diagnostics; your production check may live
120    /// next to `Dpl` to avoid cyclic deps.)
121    #[inline]
122    #[must_use]
123    pub const fn can_load_data_with_dpl(self, cpl: Ring, dpl: Ring) -> bool {
124        (self.effective_with(cpl) as u8) <= (dpl as u8)
125    }
126}
127
128impl From<Ring> for Rpl {
129    #[inline]
130    fn from(ring: Ring) -> Self {
131        Self::from_ring(ring)
132    }
133}
134
135impl From<Rpl> for Ring {
136    #[inline]
137    fn from(rpl: Rpl) -> Self {
138        rpl.to_ring()
139    }
140}

142pub const KERNEL_RPL: Rpl = Rpl::Ring0;
143pub const USER_RPL: Rpl = Rpl::Ring3;

145#[cfg(test)]
146mod tests {
147    use super::*;
148
149    #[test]
150    fn rpl_bits_roundtrip() {
151        for b in 0u16..=3 {
152            let r = Rpl::from_bits(b);
153            assert_eq!(r.into_bits(), b);
154        }
155    }
156
157    #[test]
158    fn selector_pack_unpack() {
159        let base: u16 = (0x1234 & !RPL_MASK); // pretend this is some CS without RPL
160        let with_user = USER_RPL.apply_to_selector(base);
161        assert_eq!(Rpl::from_selector(with_user), USER_RPL);
162        let with_kernel = KERNEL_RPL.apply_to_selector(base);
163        assert_eq!(Rpl::from_selector(with_kernel), KERNEL_RPL);
164        // Make sure only low 2 bits changed:
165        assert_eq!(with_user & !RPL_MASK, base & !RPL_MASK);
166        assert_eq!(with_kernel & !RPL_MASK, base & !RPL_MASK);
167    }
168
169    #[test]
170    fn build_selector_gdt_ldt() {
171        let s_gdt = USER_RPL.build_selector(5, false);
172        let s_ldt = USER_RPL.build_selector(5, true);
173        assert_eq!(Rpl::from_selector(s_gdt), USER_RPL);
174        assert_eq!(Rpl::from_selector(s_ldt), USER_RPL);
175        assert_eq!((s_gdt >> 3), 5);
176        assert_eq!((s_ldt >> 3), 5);
177        assert_eq!(((s_gdt >> 2) & 1), 0); // GDT
178        assert_eq!(((s_ldt >> 2) & 1), 1); // LDT
179    }
180
181    #[test]
182    fn effective_and_check() {
183        // CPL=0, RPL=3 → effective=3, can load only if DPL≥3
184        assert_eq!(USER_RPL.effective_with(Ring::Ring0), Ring::Ring3);
185        assert!(!USER_RPL.can_load_data_with_dpl(Ring::Ring0, Ring::Ring0));
186        assert!(USER_RPL.can_load_data_with_dpl(Ring::Ring0, Ring::Ring3));
187
188        // CPL=3, RPL=0 → effective=3 (max), same result.
189        assert_eq!(KERNEL_RPL.effective_with(Ring::Ring3), Ring::Ring3);
190    }
191}
