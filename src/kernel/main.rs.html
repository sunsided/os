29/// Stack size.
30const BOOT_STACK_SIZE: usize = 64 * 1024;
31
32/// Physical Memory mapper for the Higher-Half Direct Map (HHDM).
33static MAPPER: HhdmPhysMapper = HhdmPhysMapper;
34
35/// 16-byte aligned stack
36#[repr(align(16))]
37struct Aligned<const N: usize>([u8; N]);
38
39#[unsafe(link_section = ".bss.boot")]
40#[unsafe(no_mangle)]
41static mut BOOT_STACK: Aligned<BOOT_STACK_SIZE> = Aligned([0; BOOT_STACK_SIZE]);

43/// The kernel entry point
44///
45/// # UEFI Interaction
46/// The UEFI loader will jump here after `ExitBootServices`.
47///
48/// # ABI
49/// The ABI is defined as `sysv64` (Rust's `extern "C"`), so the kernel is called
50/// with the `boot_info` pointer in `RDI` (System V AMD64 ABI, as on Linux/x86_64).
51///
52/// # Naked function & Stack
53/// This is a naked function in order to set up the stack ourselves. Without
54/// the `naked` attribute (and the [`naked_asm`](core::arch::naked_asm) instruction), Rust
55/// compiler would apply its own assumptions based on the C ABI and would attempt to
56/// unwind the stack on the call into [`kernel_entry`]. Since we're clearing out the stack
57/// here, this would cause UB.
58#[unsafe(no_mangle)]
59#[unsafe(naked)]
60pub extern "C" fn _start_kernel(_boot_info: *const KernelBootInfo) {
61    core::arch::naked_asm!(
62        "cli",
63        // save RDI (boot_info per SysV64)
64        "mov r12, rdi",
65        // Build our own kernel stack and establish a valid call frame for kernel_entry
66        "lea rax, [rip + {stack_sym}]",
67        "add rax, {stack_size}",
68        // Align down to 16
69        "and rax, -16",
70        // Set RSP to the prepared value
71        "mov rsp, rax",
72        // Emulate a CALL by pushing a dummy return address (so RSP % 16 == 8 at entry)
73        "push 0",
74        "xor rbp, rbp",
75        // Restore boot_info into the expected arg register (SysV/C ABI)
76        "mov rdi, r12",
77        // Jump to Rust entry and never return
78        "jmp {rust_entry}",
79        stack_sym = sym BOOT_STACK,
80        stack_size = const BOOT_STACK_SIZE,
81        rust_entry = sym kernel_entry,
82    );
83}

85/// Kernel entry running on normal stack.
86///
87/// # Notes
88/// * `no_mangle` is used so that [`_start_kernel`] can jump to it by name.
89/// * It uses C ABI to have a defined convention when calling in from ASM.
90/// * The [`_start_kernel`] function keeps `boot_info` in `RDI`, matching C ABI expectations.
91#[unsafe(no_mangle)]
92extern "C" fn kernel_entry(boot_info: *const KernelBootInfo) -> ! {
93    qemu_trace!("Kernel reporting to QEMU!\n");
94
95    // Enable interrupts (undo the earlier 'cli')
96    unsafe { core::arch::asm!("sti") };
97
98    let bi = unsafe { &*boot_info };
99    trace_boot_info(bi);
100
101    let fb_virt = remap_boot_memory(bi);
102    kernel_main(&fb_virt)
103}
104
105/// Main kernel loop, running with all memory (including framebuffer) properly mapped.
106///
107/// # Memory Safety
108/// At this point, the kernel operates with virtual addresses set up by the VMM, and
109/// the framebuffer is accessible at its mapped virtual address. All further kernel
110/// code should use these mapped addresses, not physical ones.
111///
112/// # Arguments
113/// * `fb_virt` - [`FramebufferInfo`] with a valid, mapped virtual address.
114///
115/// # Safety
116/// Assumes that [`remap_boot_memory`] has been called and all required mappings are in place.
117fn kernel_main(fb_virt: &FramebufferInfo) -> ! {
118    qemu_trace!("Entering Kernel main loop ...\n");
119
120    let mut cycle = 127u8;
121    loop {
122        cycle = cycle.wrapping_add(10);
123        unsafe { fill_solid(fb_virt, 72, 0, cycle) };
124        spin_loop();
125    }
126}

128/// Remaps the boot framebuffer memory into the kernel's virtual address space.
129///
130/// UEFI provides the physical address of the framebuffer in the boot info, but does not
131/// include it in the memory mapping table. This means the kernel must manually map the
132/// framebuffer into its own virtual address space to access it. This function sets up the
133/// necessary mapping so the framebuffer can be used by the kernel.
134fn remap_boot_memory(bi: &KernelBootInfo) -> FramebufferInfo {
135    // Set up PMM (bootstrap) and VMM (kernel)
136    let mut pmm = BitmapFrameAlloc::new();
137    let mut vmm = unsafe { Vmm::from_current(&MAPPER, &mut pmm) };
138
139    // Map framebuffer
140    let fb_pa = bi.fb.framebuffer_ptr;
141    let fb_len = bi.fb.framebuffer_size;
142    let va_base = HHDM_BASE + VGA_LIKE_OFFSET;
143    let fb_flags = VirtualMemoryPageBits::default()
144        .with_writable(true)
145        .with_write_combining()
146        .with_global(true)
147        .with_no_execute(true);
148
149    vmm.map_region(
150        VirtualAddress::new(va_base),
151        PhysicalAddress::new(fb_pa),
152        fb_len,
153        fb_flags,
154        fb_flags,
155    )
156    .expect("Framebuffer mapping failed");
157
158    // Return updated FramebufferInfo with new virtual address
159    let mut fb_virt = bi.fb.clone();
160    fb_virt.framebuffer_ptr = va_base + (fb_pa & 0xFFF); // preserve offset within page
161    qemu_trace!("Remapped frame buffer to {va_base:#x}\n");
162    fb_virt
163}
