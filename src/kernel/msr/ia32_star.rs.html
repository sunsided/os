8///
9/// In 64-bit mode:
10///
11/// - `syscall` uses `IA32_STAR[47:32]` to derive kernel CS/SS.
12/// - `sysret` uses `IA32_STAR[63:48]` to derive user CS/SS.
13/// - `IA32_LSTAR` provides the 64-bit RIP target for `syscall`.
14///
15/// In compatibility mode, `IA32_STAR[31:0]` holds the 32-bit EIP target for `syscall`.
16pub trait Ia32StarExt {
17    fn from_selectors<T>(selector: T) -> Self
18    where
19        T: Borrow<Selectors>;
20
21    /// Helper to build a STAR value for a pure 64-bit kernel.
22    ///
23    /// `kernel_cs` and `user_cs` are the *selectors* (e.g. `0x08` and `0x1b`).
24    fn new_64bit(kernel_cs: SegmentSelector<CodeSel>, user_cs: SegmentSelector<CodeSel>) -> Self;
25}
26
27impl Ia32StarExt for Ia32Star {
28    fn from_selectors<T>(selector: T) -> Self
29    where
30        T: Borrow<Selectors>,
31    {
32        let selector = selector.borrow();
33        <Self as Ia32StarExt>::new_64bit(selector.kernel_cs, selector.user_cs)
34    }
35
36    /// Helper to build a STAR value for a pure 64-bit kernel.
37    ///
38    /// `kernel_cs` and `user_cs` are the *selectors* (e.g. `0x08` and `0x1b`).
39    fn new_64bit(kernel_cs: SegmentSelector<CodeSel>, user_cs: SegmentSelector<CodeSel>) -> Self {
40        // Assumption:
41        // kcode -> kdata -> udata -> ucode
42
43        // Extract and validate GDT indices
44        let kidx = kernel_cs.index();
45        let uidx = user_cs.index();
46
47        // SYSRET requires:
48        //   user_ss_index = user_cs_index - 1
49        // because SS uses (base + 8), CS uses (base + 16).
50        debug_assert_ne!(kidx, 0);
51        debug_assert_eq!(kernel_cs.rpl(), Rpl::Ring0);
52        debug_assert_ne!(uidx, 0, "User CS selector at GDT index 0 is invalid");
53
54        Self::new_64bit_raw(kernel_cs.into_bits(), user_cs.into_bits())
55    }
