1//! Virtual layout for **per-CPU kernel stacks**.
2//!
3//! Each CPU receives a fixed 1 MiB slot inside a dedicated high-half region.
4//! Within each slot we reserve one unmapped 4 KiB **guard page** at the
5//! bottom and map the actual kernel stack immediately above it.
6//!
7//! ```text
8//!  ┌──────────────────────────────┐
9//!  │   next CPU’s slot …          │
10//!  ├──────────────────────────────┤ ← BASE + (cpu_id+1)*STRIDE
11//!  │   mapped kernel stack        │
12//!  │   (RW | NX | kernel-only)    │
13//!  ├──────────────────────────────┤ ← guard + 4 KiB
14//!  │   4 KiB guard (unmapped)     │
15//!  └──────────────────────────────┘ ← slot base = BASE + cpu_id*STRIDE
16//! ```
17//!
18//! * The guard page catches stack overflows with a page-fault trap.
19//! * Each slot’s total span (`STRIDE`) leaves room for the guard + stack
20//!   and keeps adjacent CPU stacks safely separated.
21
22use kernel_info::memory::KERNEL_STACK_SIZE;
23use kernel_memory_addresses::{PageSize, Size4K, VirtualAddress, VirtualPage};
24
25/// Size of the unmapped guard page at the bottom of each stack (4 KiB).
26pub const KSTACK_GUARD: u64 = Size4K::SIZE;
27
28/// Virtual base address of the **kernel-stack region**.
29///
30/// Chosen in the canonical higher-half (kernel) address space:
31/// `0xffff_ff00_0000_0000` lies below typical higher-half kernel text/data
32/// mappings but far above user space.
33/// This keeps all per-CPU stacks in a contiguous, predictable range that
34/// doesn’t collide with identity-mapped or user regions.
35pub const KSTACK_BASE: u64 = 0xffff_ff00_0000_0000;
36
37/// Virtual span reserved per CPU (bytes).
38///
39/// Each CPU gets one contiguous 1 MiB slot for its guard + stack.
40/// 1 MiB leaves ample space even if the kernel stack grows to hundreds of KiB.
41pub const KSTACK_CPU_STRIDE: u64 = 0x10_0000; // 1 MiB per CPU
42
43const _: () = {
44    // Sanity: ensure stack size fits within the per-CPU slot.
45    assert!((KERNEL_STACK_SIZE as u64).is_multiple_of(Size4K::SIZE));
46    assert!((KERNEL_STACK_SIZE as u64) <= max_kstack_bytes());
47    assert!(KSTACK_CPU_STRIDE.is_multiple_of(Size4K::SIZE));
48};
49
50/// Maximum usable bytes for a single kernel stack (excludes guard page).
51#[inline]
52pub const fn max_kstack_bytes() -> u64 {
53    KSTACK_CPU_STRIDE - KSTACK_GUARD
54}
55
56/// Return the **guard-page base** of the kernel-stack slot for `cpu_id`.
57///
58/// The first mapped byte (stack base) lies immediately above this guard page.
59/// The top of the stack can be computed as
60/// `base + mapped_bytes`, aligned down to 16 bytes for ABI compliance.
61#[inline]
62pub const fn kstack_slot_for_cpu(cpu_id: u64) -> VirtualPage<Size4K> {
63    let addr = VirtualAddress::new(KSTACK_BASE + cpu_id * KSTACK_CPU_STRIDE);
64    let page = addr.page();
65    assert!(page.base().as_u64() == addr.as_u64());
66    page
67}
