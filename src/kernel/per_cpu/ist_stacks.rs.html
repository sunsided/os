1//! Virtual layout for IST stacks: per-CPU × per-IST slots with a guard page.
2//!
3//! Layout (virtual):
4//!   `IST_BASE`
5//!     + `cpu_id` * `CPU_STRIDE`
6//!       + (ist_idx-1) * `IST_SLOT_STRIDE`
7//!         -> [ guard (4 KiB, unmapped) ][ IST stack bytes, RW|NX ]
8//!
9//! Notes
10//! - We keep IST stacks in a separate region from kernel stacks to simplify
11//!   debugging and avoid tight packing constraints.
12//! - One unmapped 4 KiB guard below each IST stack catches overflows.
13//! - `IST_SLOT_STRIDE` must be >= guard + max IST size you’ll map.
14
15use crate::interrupts::Ist;
16use kernel_memory_addresses::{PageSize, Size4K, VirtualAddress, VirtualPage};
17
18/// Number of hardware IST entries supported by x86_64 TSS.
19pub const IST_SLOTS_PER_CPU: u64 = 7;
20
21/// 4 KiB guard below each IST stack.
22#[allow(dead_code)]
23pub const IST_GUARD: u64 = Size4K::SIZE;
24
25/// Virtual base for all IST stacks (choose a disjoint, canonical kernel range).
26/// Picked in the higher half, far from your kernel-stack region (`0xffff_ff00_…`).
27pub const IST_BASE: u64 = 0xffff_ff10_0000_0000;
28
29/// Per-CPU stride in the IST region (bytes). Leave ample room for 7 ISTs.
30pub const IST_CPU_STRIDE: u64 = 0x10_0000; // 1 MiB per CPU in the IST region
31
32/// Per-IST stride inside one CPU’s IST area (bytes).
33/// With 7 ISTs × 128 KiB < 1 MiB, this fits comfortably.
34pub const IST_SLOT_STRIDE: u64 = 0x02_0000; // 128 KiB per IST “slot”
35
36/// 16 KiB is enough for #PF/#DF handlers.
37///
38/// [`IST1`](Ist::Ist1) is used for critical handlers such as double fault.
39pub const IST1_SIZE: u64 = 16 * 1024;
40
41const _: () = {
42    // Sanity: 7 IST slots must fit inside one CPU stride
43    assert!(IST_SLOTS_PER_CPU * IST_SLOT_STRIDE <= IST_CPU_STRIDE);
44    // Page-aligned strides
45    assert!(IST_CPU_STRIDE.is_multiple_of(Size4K::SIZE));
46    assert!(IST_SLOT_STRIDE.is_multiple_of(Size4K::SIZE));
47};
48
49/// Maximum usable IST bytes per slot (excludes the 4 KiB guard).
50#[inline]
51#[allow(dead_code)]
52pub const fn max_ist_bytes() -> u64 {
53    IST_SLOT_STRIDE - IST_GUARD
54}
55
56/// Return the **guard page** base for `(cpu_id, ist_idx)` (1..=7).
57/// The first mapped byte (stack base) is `guard + 4 KiB`.
58#[inline]
59#[allow(clippy::cast_possible_truncation)]
60pub const fn ist_slot_for_cpu(cpu_id: u64, ist_idx: Ist) -> VirtualPage<Size4K> {
61    // Hardware IST indices are 1..=7.
62    assert!(ist_idx.gate_index() >= 1 && ist_idx.gate_index() <= IST_SLOTS_PER_CPU as u8);
63
64    let cpu_off = cpu_id * IST_CPU_STRIDE;
65    let ist_off = (ist_idx.gate_index() as u64 - 1) * IST_SLOT_STRIDE;
66    let addr = VirtualAddress::new(IST_BASE + cpu_off + ist_off);
67    let page = addr.page();
68    assert!(page.base().as_u64() == addr.as_u64()); // page-aligned
69    page
70}
71
72/// Convenience: compute the 16-byte–aligned top given a mapped length.
73/// You’ll typically map `[guard (4 KiB)][ist_bytes]` and then use this top.
74#[inline]
75#[allow(dead_code)]
76pub const fn ist_top_for(len_bytes_mapped: u64, guard_base: VirtualAddress) -> VirtualAddress {
77    let base = VirtualAddress::new(guard_base.as_u64() + IST_GUARD); // skip guard
78    // align down to 16 for ABI-correct entry on the stack
79    VirtualAddress::new((base.as_u64() + len_bytes_mapped) & !0xFu64)
80}
