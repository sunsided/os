6/// Minimal state for SYSCALL/SYSRET-based syscalls.
7///
8/// Layout must match the push order in the naked stub:
9///
10///   push rsp   → rsp
11///   push r11   → rflags
12///   push rcx   → rip
13///   push rdi   → arg0
14///   push rax   → sysno/ret
15///
16/// With `#[repr(C)]`, memory from top-of-stack (RSP) looks like:
17///
18///   +0  rax
19///   +8  rdi
20///   +16 rip
21///   +24 rflags
22///   +32 rsp
23#[derive(Debug)]
24#[repr(C)]
25pub struct SyscallFrame {
26    pub rax: u64, // syscall number on entry, return value on exit
27    pub rdi: u64, // arg0
28    pub rsi: u64, // arg1
29    pub rdx: u64, // arg2
30    pub r10: u64, // arg3 (normally rcx, but that's reserved by syscall)
31    pub r8: u64,  // arg4, sic!
32    pub r9: u64,  // arg5
33    pub rip: u64, // user return RIP (from RCX)
34    pub rflags: Rflags,
35    pub rsp: u64, // user stack pointer on entry
36}

38#[unsafe(naked)]
39pub extern "C" fn syscall_entry_stub() {
40    // Get the number of bytes between &PerCpu and &PerCpu.kstack_top
41    const PERCPU_KSTACK_TOP_OFFSET: usize = offset_of!(PerCpu, kstack_top);
42
43    core::arch::naked_asm!(
44        // SYSCALL entry invariants:
45        //   RCX = user RIP
46        //   R11 = user RFLAGS
47        //   RSP = user stack
48        //
49        //   RAX = syscall number
50        //   RDI = arg0
51        //   RSI = arg1
52        //   R10 = arg2
53        //   R8 = arg2
54        //   R9 = arg3
55
56        // Switch GS base to kernel PerCpu
57        "swapgs",
58
59        // Save user RSP (we’ll store it in the frame)
60        "mov rbx, rsp",
61
62        // Switch to kernel syscall stack: rsp = PerCpu.kstack_top
63        "mov rsp, qword ptr gs:[{kstack}]",
64
65        // ensure pre-call %rsp % 16 == 8 (SysV). kstack_top is 16-aligned.
66        "sub rsp, 8",
67
68        // Build SyscallFrame on kernel stack.
69        //
70        // push order (last pushed at lowest address):
71        //   rsp   (user)
72        //   rflags (user)
73        //   rip   (user)
74        //   r9    (arg5)
75        //   r8    (arg4, sic!)
76        //   r10   (arg3)
77        //   rdx   (arg2)
78        //   rsi   (arg1)
79        //   rdi   (arg0)
80        //   rax   (sysno)
81        //
82        // resulting layout at [rsp]:
83        //   +0  rax   (ret / sysno)
84        //   +8  rdi   (a0)
85        //   +16 rsi   (a1)
86        //   +24 rdx   (a2)
87        //   +32 r10   (a3)
88        //   +40 r8    (a4)
89        //   +48 r9    (a5)
90        //   +56 rip
91        //   +64 rflags
92        //   +72 rsp
93        "push rbx",   // +72: user RSP
94        "push r11",   // +64: user RFLAGS
95        "push rcx",   // +56: user RIP
96        "push r9",    // +48: a5
97        "push r8",    // +40: a4
98        "push r10",   // +32: a3
99        "push rdx",   // +24: a2
100        "push rsi",   // +16: a1
101        "push rdi",   // +8 : a0
102        "push rax",   // +0 : sysno / retval
103
104        // &SyscallFrame in RDI for Rust (SysV ABI)
105        "mov rdi, rsp",
106
107        // Call Rust dispatcher
108        "call {rust}",
109
110        // On return, Rust may have updated tf.rax. Everything else we
111        // restore exactly as the user had it.
112
113        // Load fields back into registers:
114        "mov rax, [rsp + 0]",   // return value
115        "mov rdi, [rsp + 8]",   // arg0 (restore)
116        "mov rsi, [rsp + 16]",  // arg1 (restore)
117        "mov rdx, [rsp + 24]",  // arg2 (restore)
118        "mov r10, [rsp + 32]",  // arg3 (restore)
119        "mov r8,  [rsp + 40]",  // arg4 (restore)
120        "mov r9,  [rsp + 48]",  // arg5 (restore)
121        "mov rcx, [rsp + 56]",  // user RIP
122        "mov r11, [rsp + 64]",  // user RFLAGS
123        "mov rbx, [rsp + 72]",  // user RSP
124
125        // Switch to user stack
126        "mov rsp, rbx",
127
128        // Back to user GS
129        "swapgs",
130
131        // Return to user
132        "sysretq",
133
134        kstack = const PERCPU_KSTACK_TOP_OFFSET,
135        rust = sym syscall_fast_rust,
136    );
137}

139#[allow(clippy::no_effect_underscore_binding)]
140extern "C" fn syscall_fast_rust(tf: &mut SyscallFrame) {
141    let sysno = tf.rax;
142    let a0 = tf.rdi;
143    let a1 = tf.rsi;
144    let a2 = tf.rdx;
145    let a3 = tf.r10;
146    let a4 = tf.r8; // sic!
147    let a5 = tf.r9;
148
149    tf.rax = syscall(sysno, a0, a1, a2, a3, a4, a5, SyscallSource::Syscall);
150}
