1//! # 64-bit GDT code/data descriptor encodings (typed builders)
2//!
3//! In long mode, **base** and **limit** fields of *code/data* descriptors are
4//! ignored for linear address calculation; paging provides memory protection.
5//! What **does** matter is:
6//! - **Type** (code vs data, readable/writable),
7//! - **S** (descriptor class: code/data vs system),
8//! - **DPL** (descriptor privilege level),
9//! - **P** (present),
10//! - **L** (64-bit code enable for CS),
11//! - **DB** (must be 0 for 64-bit code segments),
12//! - **G** (granularity; irrelevant here since limit is ignored).
13//!
14//! This module provides bitfield views plus a safe `Desc64` wrapper with
15//! constructors that set the correct invariants for **64-bit code** and **data**
16//! segments, so you don’t have to twiddle bits by hand.
17
18use crate::privilege::Dpl;
19use bitfield_struct::bitfield;
20
21/// Bit layout of a **64-bit code segment** descriptor.
22///
23/// Invariants enforced by `from_code_dpl`:
24/// - `typ = 0b1010` (execute + read),
25/// - `s = 1` (code/data),
26/// - `l = 1` (64-bit code),
27/// - `db = 0` (required when `l = 1`),
28/// - `p = 1` (present),
29/// - `limit`/`base` are zero (ignored by the CPU for code in long mode).
30#[bitfield(u64)]
31pub struct CodeDescBits {
32    pub limit_lo: u16, // [15:0]   (ignored in long mode)
33    pub base_lo: u16,  // [31:16]  (ignored in long mode)
34    pub base_mid: u8,  // [39:32]
35    #[bits(4)]
36    pub typ: u8, // [43:40] = 0b1010 (exec+read)
37    pub s: bool,       // [44]     = 1 (code/data)
38    #[bits(2)]
39    pub dpl: Dpl, // [46:45]  = 0 or 3
40    pub p: bool,       // [47]     = 1
41    #[bits(4)]
42    pub limit_hi: u8, // [51:48]
43    pub avl: bool,     // [52]
44    pub l: bool,       // [53]     = 1 (64-bit code)
45    pub db: bool,      // [54]     = 0 when L=1
46    pub g: bool,       // [55]
47    pub base_hi: u8,   // [63:56]
48}
49
50/// Bit layout of a **data/stack segment** descriptor.
51///
52/// Invariants enforced by `from_data_dpl`:
53/// - `typ = 0b0010` (read/write data),
54/// - `s = 1`, `l = 0`, `p = 1`,
55/// - `db` is set to `0` here (commonly set to 1 for 32-bit data, but DB has no
56///   meaning for 64-bit data segments used as DS/ES/SS in long mode).
57#[bitfield(u64)]
58pub struct DataDescBits {
59    pub limit_lo: u16, // [15:0]
60    pub base_lo: u16,  // [31:16]
61    pub base_mid: u8,  // [39:32]
62    #[bits(4)]
63    pub typ: u8, // [43:40] = 0b0010 (read/write data)
64    pub s: bool,       // [44]     = 1
65    #[bits(2)]
66    pub dpl: Dpl, // [46:45]
67    pub p: bool,       // [47]     = 1
68    #[bits(4)]
69    pub limit_hi: u8, // [51:48]
70    pub avl: bool,     // [52]
71    pub l: bool,       // [53]     = 0 for data
72    pub db: bool,      // [54]
73    pub g: bool,       // [55]
74    pub base_hi: u8,   // [63:56]
75}
76
77/// A single 8-byte GDT entry with **code** or **data** view.
78///
79/// Use the safe constructors to create valid 64-bit descriptors. The `raw` view
80/// is provided for table emission; reading the wrong structured view is UB.
81#[repr(C)]
82#[derive(Copy, Clone)]
83pub union Desc64 {
84    pub raw: u64,
85    pub code: CodeDescBits,
86    pub data: DataDescBits,
87}
88
89impl Desc64 {
90    /// Build a **64-bit code** descriptor (execute+read, `L=1`, `DB=0`).
91    ///
92    /// `dpl` must be in `0..=3` (masked internally).
93    pub const fn from_code_dpl(dpl: Dpl) -> Self {
94        let code = CodeDescBits::new()
95            .with_limit_lo(0)
96            .with_base_lo(0)
97            .with_base_mid(0)
98            .with_typ(0b1010)
99            .with_s(true)
100            .with_dpl(dpl)
101            .with_p(true)
102            .with_limit_hi(0)
103            .with_avl(false)
104            .with_l(true) // 64-bit code
105            .with_db(false) // must be 0 with L=1
106            .with_g(false)
107            .with_base_hi(0);
108        Self { code }
109    }
110
111    /// Build a **data/stack** descriptor (read/write, `L=0`).
112    ///
113    /// `dpl` must be in `0..=3` (masked internally).
114    pub const fn from_data_dpl(dpl: Dpl) -> Self {
115        let data = DataDescBits::new()
116            .with_limit_lo(0)
117            .with_base_lo(0)
118            .with_base_mid(0)
119            .with_typ(0b0010)
120            .with_s(true)
121            .with_dpl(dpl)
122            .with_p(true)
123            .with_limit_hi(0)
124            .with_avl(false)
125            .with_l(false)
126            .with_db(false)
127            .with_g(false)
128            .with_base_hi(0);
129        Self { data }
130    }
131
132    /// Raw 64-bit encoding (safe to read for either variant).
133    #[inline]
134    #[allow(dead_code)]
135    pub const fn to_u64(self) -> u64 {
136        // Reading the `raw` field is always valid.
137        unsafe { self.raw }
138    }
139
140    /// View as code bits. **Unsafe**: UB if this entry isn’t a code descriptor.
141    #[inline]
142    #[allow(dead_code)]
143    pub const unsafe fn as_code(self) -> CodeDescBits {
144        unsafe { self.code }
145    }
146
147    /// View as data bits. **Unsafe**: UB if this entry isn’t a data descriptor.
148    #[inline]
149    #[allow(dead_code)]
150    pub const unsafe fn as_data(self) -> DataDescBits {
151        unsafe { self.data }
152    }
153}
154
155// Size guards: each descriptor is exactly 8 bytes.
156const _: () = {
157    assert!(size_of::<CodeDescBits>() == 8);
158    assert!(size_of::<DataDescBits>() == 8);
159    assert!(size_of::<Desc64>() == 8);
160};
