1use bitfield_struct::bitfield;
2use kernel_memory_addresses::VirtualAddress;
3
4/// Low 8 bytes of a 64-bit *Available TSS* descriptor (type = 0x9, S=0).
5#[bitfield(u64)]
6pub struct TssDescLow {
7    pub limit_lo: u16, // [15:0]
8    pub base_lo: u16,  // [31:16]
9
10    pub base_mid: u8, // [39:32]
11    #[bits(4)]
12    pub typ: u8, // [43:40] = 0x9 (Available 64-bit TSS)
13    pub s: bool,      // [44]    = 0 (system)
14    #[bits(2)]
15    pub dpl: u8, // [46:45] typically 0
16    pub p: bool,      // [47]    = 1
17
18    #[bits(4)]
19    pub limit_hi: u8, // [51:48]
20    pub avl: bool,   // [52]    = 0
21    pub zero1: bool, // [53]    = 0 (must be 0 for system types)
22    pub zero2: bool, // [54]    = 0 (must be 0 for system types)
23    pub g: bool,     // [55] granularity (keep 0 for byte granularity)
24    pub base_hi: u8, // [63:56]
25}
26
27/// High 8 bytes of a 64-bit TSS descriptor: `base[63:32]`, reserved=0.
28#[bitfield(u64)]
29pub struct TssDescHigh {
30    pub base_upper: u32, // [31:0]  base[63:32]
31    reserved: u32,       // [63:32] must be 0
32}
33
34/// 16-byte TSS system descriptor (two consecutive GDT entries).
35#[repr(C)]
36#[derive(Copy, Clone)]
37pub struct TssDesc64 {
38    pub low: TssDescLow,
39    pub high: TssDescHigh,
40}
41
42impl TssDesc64 {
43    /// Build a 64-bit *Available TSS* descriptor (type 0x9).
44    #[inline]
45    pub const fn new(tss_base: VirtualAddress, tss_limit: u32) -> Self {
46        let limit_lo = (tss_limit & 0xFFFF) as u16;
47        let limit_hi = ((tss_limit >> 16) & 0xF) as u8;
48
49        let base_lo = (tss_base.as_u64() & 0xFFFF) as u16;
50        let base_mid = ((tss_base.as_u64() >> 16) & 0xFF) as u8;
51        let base_hi = ((tss_base.as_u64() >> 24) & 0xFF) as u8;
52        let base_upper = (tss_base.as_u64() >> 32) as u32;
53
54        let low = TssDescLow::new()
55            .with_limit_lo(limit_lo)
56            .with_base_lo(base_lo)
57            .with_base_mid(base_mid)
58            .with_typ(0x9) // Available 64-bit TSS
59            .with_s(false) // system descriptor
60            .with_dpl(0)
61            .with_p(true)
62            .with_limit_hi(limit_hi)
63            .with_avl(false)
64            .with_zero1(false)
65            .with_zero2(false)
66            .with_g(false)
67            .with_base_hi(base_hi);
68
69        let high = TssDescHigh::new().with_base_upper(base_upper);
70
71        Self { low, high }
72    }
73}
