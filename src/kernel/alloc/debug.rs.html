9#[repr(transparent)]
10#[derive(Copy, Clone)]
11struct Pte(u64);
12
13impl Pte {
14    #[inline]
15    const fn p(self) -> bool {
16        self.0 & (1 << 0) != 0
17    }
18
19    #[inline]
20    const fn rw(self) -> bool {
21        self.0 & (1 << 1) != 0
22    }
23
24    #[inline]
25    const fn us(self) -> bool {
26        self.0 & (1 << 2) != 0
27    }
28
29    #[inline]
30    const fn ps(self) -> bool {
31        self.0 & (1 << 7) != 0
32    } // page size (L3/L2)
33
34    #[inline]
35    const fn nx(self) -> bool {
36        self.0 & (1 << 63) != 0
37    } // XD/NX (leaf-relevant)
38
39    #[inline]
40    const fn addr(self) -> u64 {
41        self.0 & 0x000F_FFFF_FFFF_F000
42    }
43}

45// Raw volatile read of the idx-th u64 entry in a 4KiB table mapped via your typed view
46#[inline]
47unsafe fn read_table_u64(base_ptr: *const u64, idx: usize) -> u64 {
48    unsafe { read_volatile(base_ptr.add(idx)) }
49}

52const fn phys_to_page4k(pa: u64) -> PhysicalPage<Size4K> {
53    let base = PhysicalAddress::new(pa & !(Size4K::SIZE - 1));
54    base.page()

58pub fn dump_walk<M: PhysMapperExt>(mapper: &M, va: VirtualAddress) {
59    unsafe {
60        // Indices for VA
61        let va = va.as_u64();
62        let l4i = ((va >> 39) & 0x1FF) as usize;
63        let l3i = ((va >> 30) & 0x1FF) as usize;
64        let l2i = ((va >> 21) & 0x1FF) as usize;
65        let l1i = ((va >> 12) & 0x1FF) as usize;
66
67        // CR3 â†’ PML4 base PA
68        let mut cr3: u64;
69        core::arch::asm!("mov {}, cr3", out(reg) cr3, options(nostack, preserves_flags));
70        let pml4_pa = cr3 & 0x000F_FFFF_FFFF_F000;
71
72        // L4
73        let pml4 = core::ptr::from_mut(mapper.pml4_mut(phys_to_page4k(pml4_pa))) as *const u64;
74        let pml4e = Pte(read_table_u64(pml4, l4i));
75        info!(
76            "L4[{:3}]={:016x} P={} RW={} US={} NX={}",
77            l4i,
78            pml4e.0,
79            pml4e.p(),
80            pml4e.rw(),
81            pml4e.us(),
82            pml4e.nx()
83        );
84        if !pml4e.p() {
85            info!("-- not present at L4");
86            return;
87        }
88
89        // L3
90        let pdpt_pa = pml4e.addr();
91        let pdpt = core::ptr::from_mut(mapper.pdpt_mut(phys_to_page4k(pdpt_pa))) as *const u64;
92        let pdpte = Pte(read_table_u64(pdpt, l3i));
93        info!(
94            "L3[{:3}]={:016x} P={} RW={} US={} PS={} NX={}",
95            l3i,
96            pdpte.0,
97            pdpte.p(),
98            pdpte.rw(),
99            pdpte.us(),
100            pdpte.ps(),
101            pdpte.nx()
102        );
103        if !pdpte.p() {
104            info!("-- not present at L3");
105            return;
106        }
107        if pdpte.ps() {
108            info!("-- 1GiB leaf: US={} NX={}\n", pdpte.us(), pdpte.nx());
109            return;
110        }
111
112        // L2
113        let pd_pa = pdpte.addr();
114        let pd = core::ptr::from_mut(mapper.pd_mut(phys_to_page4k(pd_pa))) as *const u64;
115        let pde = Pte(read_table_u64(pd, l2i));
116        info!(
117            "L2[{:3}]={:016x} P={} RW={} US={} PS={} NX={}",
118            l2i,
119            pde.0,
120            pde.p(),
121            pde.rw(),
122            pde.us(),
123            pde.ps(),
124            pde.nx()
125        );
126        if !pde.p() {
127            info!("-- not present at L2");
128            return;
129        }
130        if pde.ps() {
131            info!("-- 2MiB leaf: US={} NX={}\n", pde.us(), pde.nx());
132            return;
133        }
134
135        // L1
136        let pt_pa = pde.addr();
137        let pt = core::ptr::from_mut(mapper.pt_mut(phys_to_page4k(pt_pa))) as *const u64;
138        let pte = Pte(read_table_u64(pt, l1i));
139        info!(
140            "L1[{:3}]={:016x} P={} RW={} US={} NX={}",
141            l1i,
142            pte.0,
143            pte.p(),
144            pte.rw(),
145            pte.us(),
146            pte.nx()
147        );
148    }
149}

156pub fn promote_pml4_user_bit<M: PhysMapperExt>(mapper: &M, target_va: VirtualAddress) {
157    unsafe {
158        let mut cr3: u64;
159        core::arch::asm!("mov {}, cr3", out(reg) cr3, options(nostack, preserves_flags));
160        let pml4_pa = cr3 & 0x000F_FFFF_FFFF_F000;
161        let slot = va_l4_index(target_va);
162
163        // Get a mutable view of the PML4 table
164        let pml4 = mapper.pml4_mut(PhysicalAddress::new(pml4_pa).page());
165
166        // Treat it as raw u64 entries; bit 2 is US
167        let ents: *mut u64 = core::ptr::from_mut(pml4).cast::<u64>();
168        let cur = read_volatile(ents.add(slot));
169        if (cur & 1) == 0 {
170            info!("promote_pml4_user_bit: slot {slot} not present");
171            return;
172        }
173        let new = cur | (1 << 2); // US=1
174        if new == cur {
175            info!("PML4E[{slot}] already US=1");
176        } else {
177            core::ptr::write_volatile(ents.add(slot), new);
178            info!("PML4E[{slot}]: {cur:016x} -> {new:016x} (US=1)");
179        }
180    }
181}

196pub fn clear_parent_xd_for_exec<M: PhysMapperExt>(m: &M, va: VirtualAddress) {
197    let va = va.as_u64();
198    unsafe {
199        let mut cr3: u64;
200        core::arch::asm!("mov {}, cr3", out(reg) cr3, options(nostack, preserves_flags));
201        let pml4_pa = cr3 & 0x000F_FFFF_FFFF_F000;
202
203        let pml4 =
204            core::ptr::from_mut(m.pml4_mut(PhysicalAddress::new(pml4_pa).page())).cast::<u64>();
205        let pe4 = pml4.add(l4i(va));
206        let e4 = core::ptr::read_volatile(pe4);
207        let pdpt_pa = e4 & 0x000F_FFFF_FFFF_F000;
208
209        let pdpt =
210            core::ptr::from_mut(m.pdpt_mut(PhysicalAddress::new(pdpt_pa).page())).cast::<u64>();
211        let pe3 = pdpt.add(l3i(va));
212        let mut e3 = core::ptr::read_volatile(pe3);
213        if e3 & (1 << 7) == 0 {
214            // not a 1GiB leaf
215            // clear NX at L3
216            if e3 & (1u64 << 63) != 0 {
217                e3 &= !(1u64 << 63);
218                core::ptr::write_volatile(pe3, e3);
219            }
220            let pd_pa = e3 & 0x000F_FFFF_FFFF_F000;
221
222            let pd =
223                core::ptr::from_mut(m.pd_mut(PhysicalAddress::new(pd_pa).page())).cast::<u64>();
224            let pe2 = pd.add(l2i(va));
225            let mut e2 = core::ptr::read_volatile(pe2);
226            // if 2MiB leaf, also clear NX here; else just as parent
227            if e2 & (1u64 << 63) != 0 {
228                e2 &= !(1u64 << 63);
229                core::ptr::write_volatile(pe2, e2);
230            }
231        }
232    }
