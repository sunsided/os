1/// Hardware **Interrupt Stack Table (IST)** selector.
2///
3/// x86-64 TSS defines up to seven IST pointers (`IST[0]..IST[6]`).
4/// Each IDT gate may specify `.ist(n)` (1–7) to make the CPU load
5/// `TSS.IST[n-1]` as the new stack pointer when that interrupt fires.
6///
7/// Index 0 in the *IDT gate* means “no IST” (use the current stack).
8#[derive(Debug, Clone, Copy, PartialEq, Eq)]
9#[repr(u8)]
10pub enum Ist {
11    /// No alternate stack (IDT `.ist(0)`).
12    None = 0,
13
14    /// Hardware IST#1 → `TSS.IST[0]`.
15    Ist1 = 1,
16    /// Hardware IST#2 → `TSS.IST[1]`.
17    Ist2 = 2,
18    /// Hardware IST#3 → `TSS.IST[2]`.
19    Ist3 = 3,
20    /// Hardware IST#4 → `TSS.IST[3]`.
21    Ist4 = 4,
22    /// Hardware IST#5 → `TSS.IST[4]`.
23    Ist5 = 5,
24    /// Hardware IST#6 → `TSS.IST[5]`.
25    Ist6 = 6,
26    /// Hardware IST#7 → `TSS.IST[6]`.
27    Ist7 = 7,
28}
29
30impl Ist {
31    #[inline]
32    pub const fn from_bits(value: u8) -> Self {
33        match value {
34            1 => Self::Ist1,
35            2 => Self::Ist2,
36            3 => Self::Ist3,
37            4 => Self::Ist4,
38            5 => Self::Ist5,
39            6 => Self::Ist6,
40            7 => Self::Ist7,
41            _ => Self::None,
42        }
43    }
44
45    #[inline]
46    pub const fn into_bits(self) -> u8 {
47        self as u8
48    }
49
50    /// Returns the 0-based array index into `TSS.ist[]`.
51    ///
52    /// Returns `None` if this variant is [`Ist::None`].
53    #[inline]
54    pub const fn tss_index(self) -> Option<usize> {
55        match self {
56            Self::None => None,
57            Self::Ist1 => Some(0),
58            Self::Ist2 => Some(1),
59            Self::Ist3 => Some(2),
60            Self::Ist4 => Some(3),
61            Self::Ist5 => Some(4),
62            Self::Ist6 => Some(5),
63            Self::Ist7 => Some(6),
64        }
65    }
66
67    /// Return the raw numeric value to write into an IDT gate’s `.ist()` field.
68    #[inline]
69    pub const fn gate_index(self) -> u8 {
70        self.into_bits()
71    }
72}
