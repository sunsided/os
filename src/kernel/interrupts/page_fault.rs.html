12pub const PAGE_FAULT_VECTOR: usize = 0x0E; // 14
13
14pub trait PageFaultInterrupt {
15    /// Install a ring-0 page-fault handler (no user-call) and assign an IST slot.
16    fn init_page_fault_gate_ist(&mut self, handler: extern "C" fn(), ist: Ist) -> &mut Self;
17}
18
19impl PageFaultInterrupt for Idt {
20    fn init_page_fault_gate_ist(&mut self, handler: extern "C" fn(), ist: Ist) -> &mut Self {
21        self[PAGE_FAULT_VECTOR]
22            .set_handler(handler)
23            .selector(KERNEL_CS_SEL)
24            .present(true)
25            .ist(ist)
26            .gate_type(GateType::InterruptGate);
27        self
28    }
29}

31/// Interrupt-gate PF handler: reads CR2 and the pushed error code, then halts.
32///
33/// # Safety
34/// Early-bringup only. Does not attempt to resume execution.
35#[unsafe(naked)]
36pub extern "C" fn page_fault_handler() {
37    naked_asm!(
38        "cli",
39        // Save a minimal set of caller-saved regs we’ll use (SysV: rax, rcx, rdx, rsi, rdi, r8-r11).
40        "push rax",
41        "push rdi",
42        "push rsi",
43
44        // ENTRY swapgs if from CPL3: CS at [rsp + 40]
45        "mov rax, [rsp + 40]",
46        "test al, 3",
47        "jz 1f",
48        "swapgs",
49        "1:",
50
51        // rdi := cr2 (first arg)
52        "mov rdi, cr2",
53        // The CPU pushed an error code before entering the handler.
54        // We just pushed 3 regs → error code is now at [rsp + 3*8].
55        "mov rsi, [rsp + 24]",   // rsi := error code (second arg)
56        "call {log_pf}",         // log_page_fault(cr2, err)
57
58        // Stop here; don't try to return in early bringup.
59        "1: hlt",
60        "jmp 1b",
61        log_pf = sym log_page_fault
62    )
63}

112/// Page-fault error code layout (x86-64).
113///
114/// Each bit describes the condition that caused the page fault.
115/// Reference: Intel SDM Vol. 3A, §6.15.1 “Page-Fault Exception (#PF)”.
116#[bitfield(u64)]
117pub struct PageFaultError {
118    /// 0 = non-present page.
119    /// 1 = protection violation (page present but access disallowed).
120    pub present: bool, // bit 0
121
122    /// 0 = read or execute.
123    /// 1 = write access.
124    pub write: bool, // bit 1
125
126    /// 0 = supervisor (CPL 0–2).
127    /// 1 = user mode (CPL 3).
128    pub user: bool, // bit 2
129
130    /// 1 = caused by reserved bit set in a paging structure.
131    pub reserved_bit: bool, // bit 3
132
133    /// 1 = instruction fetch (execute access).
134    pub instruction_fetch: bool, // bit 4
135
136    /// 1 = protection-key violation (if CR4.PKE=1).
137    pub protection_key: bool, // bit 5
138
139    /// 1 = shadow stack access (if CET-SS enabled).
140    pub shadow_stack: bool, // bit 6
141
142    #[bits(57)]
143    __: u64, // reserved / ignored bits
144}
145
146impl PageFaultError {
147    pub const fn explain(self) -> &'static str {
148        if !self.present() {
149            "Non-present page (page not mapped or swapped out)"
150        } else if self.instruction_fetch() {
151            if self.user() {
152                "User-mode instruction fetch on protected page (likely NX or SMEP)"
153            } else {
154                "Kernel instruction fetch on protected page"
155            }
156        } else if self.write() {
157            "Write access to protected page"
158        } else {
159            "Read access to protected page"
160        }
161    }
