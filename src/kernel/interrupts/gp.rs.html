8pub const GP_FAULT_VECTOR: usize = 0x0D; // 13
9
10pub trait GeneralProtectionFaultInterrupt {
11    fn init_gp_fault_gate(&mut self, handler: extern "C" fn()) -> &mut Self;
12}
13
14impl GeneralProtectionFaultInterrupt for Idt {
15    fn init_gp_fault_gate(&mut self, handler: extern "C" fn()) -> &mut Self {
16        self[GP_FAULT_VECTOR]
17            .set_handler(handler)
18            .selector(KERNEL_CS_SEL)
19            .present(true)
20            .user_callable()
21            .gate_type(GateType::InterruptGate);
22        self
23    }
24}

26#[unsafe(naked)]
27pub extern "C" fn gp_fault_handler() {
28    naked_asm!(
29        "cli",
30        // Save a minimal set of caller-saved regs we’ll use (SysV: rax, rdi, rsi)
31        "push rax",
32        "push rdi",
33        "push rsi",
34
35        // ENTRY swapgs if from CPL3: CS at [rsp + 40]
36        "mov rax, [rsp + 40]",
37        "test al, 3",
38        "jz 1f",
39        "swapgs",
40        "1:",
41
42        // rdi := RIP (first arg)
43        "mov rdi, [rsp + 32]",   // 3 pushes * 8 + 8 = 24 + 8 = 32
44        // rsi := error code (second arg)
45        "mov rsi, [rsp + 24]",   // 3 pushes * 8 = 24
46
47        "call {log_gp}",
48
49        // Park the CPU
50        "1: hlt",
51        "jmp 1b",
52
53        log_gp = sym log_gp_fault
54    )
55}

95#[derive(Debug, Clone, Copy)]
96struct GpErrorInfo {
97    selector_index: u16,
98    ti_ldt: bool,
99    external: bool,
100}
101
102const fn decode_gp_error(err: u64) -> GpErrorInfo {
103    // If you want, treat zero specially; here we always return Some.
104    GpErrorInfo {
105        selector_index: ((err >> 3) & 0x1fff) as u16, // bits 15:3
106        ti_ldt: ((err >> 2) & 0x1) != 0,              // Table Indicator: 0=GDT, 1=LDT
107        external: (err & 0x1) != 0,                   // EXT bit
108    }
109}
