1//! x86_64 INT 0x80 syscall entry and minimal dispatcher.
2//!
3//! ## Overview
4//! We expose a classic Linux-style **software interrupt** (vector `0x80`) as the
5//! user→kernel syscall path. A `#[naked]` asm stub:
6//! 1) Saves all GP registers into a contiguous `TrapFrame` on the current stack,
7//! 2) passes `&mut TrapFrame` to a small Rust dispatcher,
8//! 3) restores registers, then issues `iretq` to resume the interrupted context.
9//!
10//! ## ABI used **by this kernel**
11//! - `RAX` = syscall number (`Sysno`).
12//! - `RDI`, `RSI`, `RDX` = first three integer args (`a0..a2`).
13//! - Return value is written back to `RAX` by the handler.
14//!
15//! ## Integration requirements
16//! - IDT entry for vector **0x80** must be a **64-bit interrupt gate** (type=0xE),
17//!   `P=1`, **DPL=3** (user-callable), and point to `syscall_int80_handler`.
18//! - A valid **TSS** must be loaded (`ltr`) with `rsp0` set. If you route 0x80 via
19//!   an **IST**, set the IST index in the gate accordingly.
20//! - If you rely on `GS`-based per-CPU data in kernel mode, you likely need a
21//!   `swapgs` on entry/exit (not shown here).
22//!
23//! ## Error model
24//! Unknown/unsupported syscalls return `u64::MAX` (a stand-in for `-ENOSYS`).
25//!
26//! ## Safety / portability notes
27//! - This is x86_64-only and uses a `#[naked]` function with inline asm.
28//! - We rely on the precise **push order** below; keep `TrapFrame` in sync.
29//! - Stack alignment: on entry the CPU pushes 5×8 bytes; we then push 15×8,
30//!   so total push is **160 bytes**, preserving 16-byte alignment before the call
31//!   into Rust (SysV ABI-friendly).
32//! - The IF flag is **cleared** on interrupt-gate entry. We do not re-enable
33//!   interrupts inside the stub.
34//!
35//! ## Frame shape (top of stack → bottom):
36//!   rax, rbx, rcx, rdx, rsi, rdi, rbp, r8, r9, r10, r11, r12, r13, r14, r15,
37//!   then CPU-pushed interrupt frame: rip, cs, rflags, rsp, ss.
38//!
39//! `rsp` at the call site points at `rax` ⇒ `&TrapFrame` starts at `rax`.
40

43use crate::gdt::KERNEL_CS_SEL;
44use crate::interrupts::{GateType, Idt};

48pub const SYSCALL_VECTOR: usize = 0x80; // 128
49
50pub trait SyscallInterrupt {
51    fn init_syscall_gate(&mut self) -> &mut Self;
52}
53
54impl SyscallInterrupt for Idt {
55    fn init_syscall_gate(&mut self) -> &mut Self {
56        self[SYSCALL_VECTOR]
57            .set_handler(syscall_int80_handler)
58            .selector(KERNEL_CS_SEL)
59            .present(true)
60            .user_callable()
61            .gate_type(GateType::InterruptGate);
62        self
63    }
64}

66/// Saved register/interrupt context for an INT 0x80 syscall.
67///
68/// **Layout must match the push order in the naked stub.**
69#[derive(Debug)]
70#[repr(C)]
71pub struct TrapFrame {
72    // Pushed by the stub before calling Rust handler:
73    rax: u64,
74    rbx: u64,
75    rcx: u64,
76    rdx: u64,
77    rsi: u64,
78    rdi: u64,
79    rbp: u64,
80    r8: u64,
81    r9: u64,
82    r10: u64,
83    r11: u64,
84    r12: u64,
85    r13: u64,
86    r14: u64,
87    r15: u64,
88
89    // TODO: Validate the CPU frame is indeed in the correct location here.
90
91    // Pushed by CPU on interrupt gate entry (x86_64):
92    // Order: RIP, CS, RFLAGS, RSP, SS
93    /// Return RIP saved by the CPU.
94    rip: u64,
95    /// Code segment selector at the time of INT 0x80 (usually user CS).
96    cs: u64,
97    /// RFLAGS at the time of INT 0x80 (IF is cleared on entry).
98    rflags: Rflags,
99    /// Return RSP saved by the CPU (user stack pointer on entry).
100    rsp: u64,
101    /// Stack segment selector (user SS).
102    ss: u64,
103}

105#[unsafe(naked)]
106extern "C" fn syscall_int80_handler() {
107    // Naked: no compiler prologue/epilogue; we must be perfectly balanced.
108    core::arch::naked_asm!(
109        // --- Callee-save and caller-save alike: we capture the full GPR set. ---
110        // Push order defines our TrapFrame order (top → bottom).
111        "push r15",
112        "push r14",
113        "push r13",
114        "push r12",
115        "push r11",
116        "push r10",
117        "push r9",
118        "push r8",
119        "push rbp",
120        "push rdi",
121        "push rsi",
122        "push rdx",
123        "push rcx",
124        "push rbx",
125        "push rax",
126
127        // ENTRY: if came from CPL3, swapgs to kernel GS base ---
128        "mov rax, [rsp + 128]",    // saved CS
129        "test al, 3",
130        "jz  1f",
131        "swapgs",
132        "1:",
133
134        // At this point, [rsp] = rax field ⇒ &TrapFrame == rsp.
135        // SysV: first arg in RDI.
136        "mov rdi, rsp",
137
138        // Call into Rust. We kept stack 16-byte aligned (160 bytes pushed).
139        // The Rust side writes the return value back to tf.rax.
140        "call {rust}",
141
142        // EXIT: if returning to CPL3, swapgs back to user GS base ---
143        // Do this BEFORE popping regs so any scratch doesn’t leak to user
144        "mov rax, [rsp + 128]",    // saved CS for return path
145        "test al, 3",
146        "jz  1f",
147        "swapgs",
148        "1:",
149
150        // Restore in strict reverse order.
151        "pop rax",
152        "pop rbx",
153        "pop rcx",
154        "pop rdx",
155        "pop rsi",
156        "pop rdi",
157        "pop rbp",
158        "pop r8",
159        "pop r9",
160        "pop r10",
161        "pop r11",
162        "pop r12",
163        "pop r13",
164        "pop r14",
165        "pop r15",
166
167        "iretq",
168
169        rust = sym syscall_int80_rust
170    )
171}

173/// Rust dispatcher invoked by the low-level entry stub.
174///
175/// Contract:
176/// - Reads `sysno` from `tf.rax`, args from `tf.{rdi,rsi,rdx}` (kernel ABI).
177/// - Writes the return value to `tf.rax`.
178/// - Must not assume interrupts are enabled; they are not.
179#[allow(clippy::no_effect_underscore_binding)]
180extern "C" fn syscall_int80_rust(tf: &mut TrapFrame) {
181    let sysno = tf.rax;
182    let a0 = tf.rdi;
183    let a1 = tf.rsi;
184    let a2 = tf.rdx;
185    let a3 = tf.r10;
186    let a4 = tf.r8; // sic!
187    let a5 = tf.r9;
188
189    tf.rax = syscall(sysno, a0, a1, a2, a3, a4, a5, SyscallSource::Int80h);
190}
