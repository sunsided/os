1//! # Minimal 64-bit Task State Segment (TSS) for x86_64
2//!
3//! ## What is the TSS for in 64-bit mode?
4//! In 64-bit mode, the CPU no longer performs hardware task switching, but it
5//! **still uses the TSS** to find safe stacks when transitioning from user
6//! space (Ring-3) to the kernel (Ring-0), and optionally for **IST** stacks.
7//!
8//! - `rsp0` is the **Ring-0 stack** used automatically on privilege transitions
9//!   (e.g., `int 0x80`, page faults, syscalls via interrupt gates).
10//! - `ist1..ist7` are optional **Interrupt Stack Table** entries that you can
11//!   bind to specific IDT entries to guarantee a clean stack for critical
12//!   faults (like double fault) even if your normal kernel stack is corrupted.
13//! - `iopb_offset` points to the I/O permission bitmap. Setting it to the
14//!   **end of the TSS** disables the bitmap (typical for kernels).
15//!
16//! ## How we use it
17//! - `init_tss(rsp0, ist1)` fills a global TSS with a kernel stack and
18//!   optional IST1. You will reference this TSS from the GDT's **TSS
19//!   descriptor**, then load it into the **Task Register** via `ltr`.
20//! - Later, you can change `rsp0` via `set_rsp0()`.
21//!
22//! For SMP, create one TSS per CPU and load the CPU-local TSS on AP startup.
23

28/// 64-bit Task State Segment (TSS) as used by x86-64 long mode.
29///
30/// In 64-bit mode, the CPU no longer performs hardware task switches, but it
31/// **still consults the TSS** for two critical things:
32///
33/// 1) **Privilege Stack Table (PST)** — the `rsp0..rsp2` pointers.
34///    When the processor enters the kernel through an **interrupt/exception/trap
35///    gate** that causes a **privilege-level change** (e.g., Ring-3 → Ring-0 on
36///    `int 0x80`, page fault from userland, etc.), it loads the new stack
37///    pointer from the PST entry corresponding to the target CPL (for Ring-0
38///    transitions, `rsp0`). This guarantees entry on a **kernel-controlled**
39///    stack even if userland’s RSP is bogus.
40///
41/// 2) **Interrupt Stack Table (IST)** — the `ist1..ist7` pointers.
42///    If the IDT gate used to deliver an interrupt/exception specifies a **non-zero
43///    IST index** (1..7), the processor will ignore the normal PST and instead
44///    switch to the specified **IST stack** before pushing the interrupt frame.
45///    This is essential for handling cases like **double faults** safely (when
46///    your current stack may be corrupted).
47///
48/// The **I/O Permission Bitmap** (controlled via `iopb_offset`) optionally
49/// restricts `in`/`out` port I/O for tasks at CPL > IOPL. Most kernels either
50/// **disable** the bitmap for the kernel or place a bitmap at the end of the TSS
51/// to **deny all** user I/O by default.
52///
53/// Notes:
54/// - All `_reserved*` fields must be zero.
55/// - The TSS can reside at any (canonical) address; the GDT holds a 16-byte TSS
56///   system descriptor that points to it.
57/// - If you **don’t** use an IST for an IDT entry, the CPU will use the PST
58///   (`rsp0..2`) when there is a privilege change; otherwise it continues on the
59///   current stack.
60/// - `syscall/sysret` do **not** consult the TSS; they use MSRs (`STAR/LSTAR/SFMASK`)
61///   and usually `swapgs` with a per-CPU stack scheme. Interrupt-gate-based
62///   syscalls (e.g., `int 0x80`) **do** use the TSS.
63///
64#[repr(C, packed)]
65pub struct Tss64 {
66    /// Must be zero. Reserved by the architecture.
67    ///
68    /// The first 4 bytes of the 64-bit TSS layout are reserved to align the
69    /// following `rsp0` field on an 8-byte boundary.
70    _reserved0: u32,
71
72    /// **Ring-0 stack pointer (RSP0)** used on privilege elevation.
73    ///
74    /// When the CPU delivers an interrupt/exception/trap to a gate that changes
75    /// CPL to 0 (e.g., user → kernel), it **loads RSP from `rsp0`** and then
76    /// pushes the saved user context on this kernel stack before entering the
77    /// handler. This is how you ensure a trustworthy stack on kernel entry.
78    ///
79    /// Typical kernel setup: point this to the **top of a kernel stack** that
80    /// grows downward. In SMP systems you usually set a **different `rsp0` per
81    /// CPU** (i.e., one TSS per CPU).
82    pub rsp0: VirtualAddress,
83
84    /// **Ring-1 stack pointer (RSP1)** — rarely used in modern OSes.
85    ///
86    /// If you ever use CPL=1 code segments (uncommon), a privilege transition
87    /// that targets CPL=1 would load RSP from here.
88    pub rsp1: VirtualAddress,
89
90    /// **Ring-2 stack pointer (RSP2)** — rarely used in modern OSes.
91    ///
92    /// Analogous to `rsp1`, but for transitions targeting CPL=2.
93    pub rsp2: VirtualAddress,
94
95    /// Must be zero. Reserved by the architecture.
96    _reserved1: u64,
97
98    /// **IST1**: optional dedicated stack for an IDT entry.
99    ///
100    /// If an IDT gate has its IST index field set to 1, the CPU loads RSP from
101    /// this field **before** pushing the interrupt frame, ensuring the handler
102    /// runs on this stack **regardless of privilege level**. Use for critical
103    /// handlers (e.g., **double fault**) to avoid recursion/corruption from a
104    /// failing main stack.
105    pub ist1: VirtualAddress,
106
107    /// **IST2**: optional dedicated stack for an IDT entry (IST index = 2).
108    pub ist2: VirtualAddress,
109
110    /// **IST3**: optional dedicated stack for an IDT entry (IST index = 3).
111    pub ist3: VirtualAddress,
112
113    /// **IST4**: optional dedicated stack for an IDT entry (IST index = 4).
114    pub ist4: VirtualAddress,
115
116    /// **IST5**: optional dedicated stack for an IDT entry (IST index = 5).
117    pub ist5: VirtualAddress,
118
119    /// **IST6**: optional dedicated stack for an IDT entry (IST index = 6).
120    pub ist6: VirtualAddress,
121
122    /// **IST7**: optional dedicated stack for an IDT entry (IST index = 7).
123    pub ist7: VirtualAddress,
124
125    /// Must be zero. Reserved by the architecture.
126    _reserved2: u64,
127
128    /// Must be zero. Reserved by the architecture.
129    _reserved3: u16,
130
131    /// Byte offset from the **base of this TSS** to the start of the **I/O Permission Bitmap**.
132    ///
133    /// - If this **offset is greater than or equal to the TSS limit**, then
134    ///   **no I/O bitmap is present**. In that case, access to `in/out`
135    ///   instructions is governed solely by **IOPL vs CPL** (typical kernels run
136    ///   with IOPL=0 and user CPL=3, so userland port I/O will fault).
137    /// - If present, each **bit** corresponds to one I/O port: a set bit
138    ///   generally **disallows** access to that port when CPL > IOPL; a clear bit
139    ///   **allows** it. The bitmap must be **byte-aligned** and, per manuals,
140    ///   should be followed by a **guard/terminator byte** with all bits set.
141    ///
142    /// Most kernels:
143    /// - set this to `core::mem::size_of::<Tss64>() as u16` to **disable** the
144    ///   bitmap for the kernel TSS, or
145    /// - place a bitmap at the end and fill it with `0xFF` to **deny all** user
146    ///   port I/O by default (granting specific ports only if needed).
147    pub iopb_offset: u16,
148}
149
150impl Tss64 {
151    #[allow(clippy::cast_possible_truncation)]
152    pub const fn new() -> Self {
153        Self {
154            _reserved0: 0,
155            rsp0: VirtualAddress::zero(),
156            rsp1: VirtualAddress::zero(),
157            rsp2: VirtualAddress::zero(),
158            _reserved1: 0,
159            ist1: VirtualAddress::zero(),
160            ist2: VirtualAddress::zero(),
161            ist3: VirtualAddress::zero(),
162            ist4: VirtualAddress::zero(),
163            ist5: VirtualAddress::zero(),
164            ist6: VirtualAddress::zero(),
165            ist7: VirtualAddress::zero(),
166            _reserved2: 0,
167            _reserved3: 0,
168            iopb_offset: size_of::<Self>() as u16,
169        }
170    }
171}

173/// Initialize the TSS with a kernel RSP0 and optional IST1.
174///
175/// * `kernel_stack_top` — top (highest address) of a valid kernel stack.
176///   The CPU switches to this when entering Ring-0 from Ring-3 via an
177///   interrupt/exception gate (e.g., `int 0x80`).
178/// * `ist1_top` — top of an IST1 stack for critical handlers (you
179///   later bind an IDT entry to IST1).
180pub const fn init_tss(p: &mut PerCpu, kernel_stack_top: VirtualAddress, ist1_top: VirtualAddress) {
181    let tss = &mut p.tss;
182
183    tss.rsp0 = kernel_stack_top;
184    tss.ist1 = ist1_top;
185}
186
187/// Update the Ring-0 stack pointer used on user→kernel transitions.
188#[allow(dead_code)]
189pub const fn set_rsp0(p: &mut PerCpu, new_top: VirtualAddress) {
190    p.tss.rsp0 = new_top;
191}
