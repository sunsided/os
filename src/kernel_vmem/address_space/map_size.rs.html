22/// # Page-size–directed mapping behavior
23///
24/// `MapSize` encodes, at the type level, how to:
25/// 1) **ensure** the non-leaf page-table chain exists for a given virtual
26///    address, and
27/// 2) **install** the correct **leaf** entry for that page size.
28///
29/// Implementations for [`Size1G`], [`Size2M`], and [`Size4K`] decide **where to
30/// stop the walk** and **which entry to write**, so callers don’t branch at
31/// runtime. This keeps the mapping code zero-cost and compile-time checked.
32///
33/// ## What `ensure_chain_for` returns
34///
35/// It returns the **target table frame (4 KiB page)** into which you will write
36/// the leaf entry for `Self`:
37///
38/// - For **1 GiB** pages (`Self = Size1G`): returns the **PDPT** frame
39///   *(you will write a PDPTE with `PS=1`)*.
40/// - For **2 MiB** pages (`Self = Size2M`): returns the **PD** frame
41///   *(you will write a PDE with `PS=1`)*.
42/// - For **4 KiB** pages (`Self = Size4K`): returns the **PT** frame
43///   *(you will write a PTE with `PS=0`)*.
44///
45/// Newly created non-leaf entries are initialized with `nonleaf_flags`
46/// (e.g., `present | writable`), and any **conflicting huge leaves are split**
47/// on demand by allocating and linking the next-level table.
48///
49/// ## Typical flow
50///
51/// ```ignore
52/// // Decide size with the type parameter S, no runtime branching:
53/// let leaf_table = S::ensure_chain_for(aspace, alloc, va, nonleaf_flags)?;
54/// S::set_leaf(aspace, leaf_table, va, pa, leaf_flags);
55/// ```
56///
57/// ## Safety & alignment
58///
59/// - Physical alignment is **asserted (debug)** by callers via `pa.offset::<S>() == 0`.
60/// - The mapper (`PhysMapper`) must yield **writable** views of table frames.
61/// - If you mutate the **active** address space, perform the required **TLB
62///   maintenance** (`invlpg` per page or CR3 reload).
63pub trait MapSize: PageSize {
64    /// Ensure that the non-leaf chain for `va` exists down to the table that
65    /// holds the **leaf** for `Self`, allocating and linking intermediate
66    /// tables as needed.
67    ///
68    /// ### Returns
69    /// The 4 KiB **frame** (as `PhysicalPage<Size4K>`) of the table where the
70    /// leaf for `Self` must be written:
71    /// - `Size1G` → PDPT frame
72    /// - `Size2M` → PD frame
73    /// - `Size4K` → PT frame
74    ///
75    /// ### Behavior
76    /// - Initializes newly allocated non-leaf tables to zeroed state and links
77    ///   them with `nonleaf_flags`.
78    /// - If a conflicting huge leaf is encountered at a higher level, it is
79    ///   **split** by allocating the next-level table and relinking.
80    ///
81    /// ### Errors
82    /// - `"oom: pdpt" / "oom: pd" / "oom: pt"` if allocating an intermediate
83    ///   table frame fails.
84    fn ensure_chain_for<A: PhysFrameAlloc, M: PhysMapper>(
85        aspace: &AddressSpace<M>,
86        alloc: &mut A,
87        va: VirtualAddress,
88        nonleaf_flags: VirtualMemoryPageBits,
89    ) -> Result<PhysicalPage<Size4K>, MapSizeEnsureChainError>;
90
91    /// Install the **leaf** entry for `va → pa` in the `leaf_tbl_page`
92    /// returned by [`ensure_chain_for`](Self::ensure_chain_for), with the given `leaf_flags`.
93    ///
94    /// - `Size1G`: writes a **PDPTE (PS=1)** into the PDPT at `va`.
95    /// - `Size2M`: writes a **PDE   (PS=1)** into the PD   at `va`.
96    /// - `Size4K`: writes a **PTE   (PS=0)** into the PT   at `va`.
97    ///
98    /// Callers should assert (in debug) that `pa` is aligned to `Self`:
99    /// `debug_assert_eq!(pa.offset::<Self>().as_u64(), 0)`.
100    fn set_leaf<M: PhysMapper>(
101        aspace: &AddressSpace<M>,
102        leaf_tbl_page: PhysicalPage<Size4K>,
103        va: VirtualAddress,
104        pa: PhysicalAddress,
105        leaf_flags: VirtualMemoryPageBits,
106    );
107}

109/// Error returned by [`MapSize::ensure_chain_for`] when allocating a new
110/// intermediate table frame fails.
111#[derive(Debug, Copy, Clone, PartialEq, Eq, thiserror::Error)]
112#[allow(clippy::enum_variant_names)]
113pub enum MapSizeEnsureChainError {
114    #[error("out of memory (PDPT)")]
115    OomPdpt,
116    #[error("out of memory (PD)")]
117    OomPd,
118    #[error("out of memory (PT)")]
119    OomPt,
120}

122impl MapSize for Size1G {
123    fn ensure_chain_for<A: PhysFrameAlloc, M: PhysMapper>(
124        aspace: &AddressSpace<M>,
125        alloc: &mut A,
126        va: VirtualAddress,
127        nonleaf_flags: VirtualMemoryPageBits,
128    ) -> Result<PhysicalPage<Size4K>, MapSizeEnsureChainError> {
129        let i4 = L4Index::from(va);
130
131        // L4 → L3
132        let pml4 = aspace.pml4_mut();
133        let e4 = pml4.get(i4);
134        if let Some(pdpt_page) = e4.next_table() {
135            return Ok(pdpt_page);
136        }
137
138        let f = alloc.alloc_4k().ok_or(MapSizeEnsureChainError::OomPdpt)?;
139        aspace.zero_pdpt(f);
140        pml4.set(i4, Pml4Entry::present_with(nonleaf_flags, f));
141        Ok(f)
142    }
143
144    fn set_leaf<M: PhysMapper>(
145        aspace: &AddressSpace<M>,
146        leaf_tbl_page: PhysicalPage<Size4K>,
147        va: VirtualAddress,
148        pa: PhysicalAddress,
149        leaf_flags: VirtualMemoryPageBits,
150    ) {
151        // require 1 GiB alignment in debug
152        debug_assert_eq!(pa.offset::<Self>().as_u64(), 0);
153        let pdpt = aspace.pdpt_mut(leaf_tbl_page);
154        let idx = L3Index::from(va);
155        let g1 = PhysicalPage::<Self>::from_addr(pa);
156        pdpt.set(idx, PdptEntry::present_leaf_with(leaf_flags, g1));
157    }
158}

160impl MapSize for Size2M {
161    fn ensure_chain_for<A: PhysFrameAlloc, M: PhysMapper>(
162        aspace: &AddressSpace<M>,
163        alloc: &mut A,
164        va: VirtualAddress,
165        nonleaf_flags: VirtualMemoryPageBits,
166    ) -> Result<PhysicalPage<Size4K>, MapSizeEnsureChainError> {
167        let i4 = L4Index::from(va);
168        let i3 = L3Index::from(va);
169
170        // L4 → L3
171        let pml4 = aspace.pml4_mut();
172        let e4 = pml4.get(i4);
173        let pdpt_page = if let Some(p) = e4.next_table() {
174            p
175        } else {
176            let f = alloc.alloc_4k().ok_or(MapSizeEnsureChainError::OomPdpt)?;
177            aspace.zero_pdpt(f);
178            pml4.set(i4, Pml4Entry::present_with(nonleaf_flags, f));
179            f
180        };
181
182        // L3 → L2 (and split 1GiB if necessary)
183        let pdpt = aspace.pdpt_mut(pdpt_page);
184        let e3 = pdpt.get(i3);
185        Ok(match e3.kind() {
186            Some(PdptEntryKind::NextPageDirectory(pd, _)) => pd,
187            Some(PdptEntryKind::Leaf1GiB(_, _)) | None => {
188                let f = alloc.alloc_4k().ok_or(MapSizeEnsureChainError::OomPd)?;
189                aspace.zero_pd(f);
190                pdpt.set(i3, PdptEntry::present_next_with(nonleaf_flags, f));
191                f
192            }
193        })
194    }
195
196    fn set_leaf<M: PhysMapper>(
197        aspace: &AddressSpace<M>,
198        leaf_tbl_page: PhysicalPage<Size4K>,
199        va: VirtualAddress,
200        pa: PhysicalAddress,
201        leaf_flags: VirtualMemoryPageBits,
202    ) {
203        debug_assert_eq!(pa.offset::<Self>().as_u64(), 0);
204        let pd = aspace.pd_mut(leaf_tbl_page);
205        let idx = L2Index::from(va);
206        let m2 = PhysicalPage::<Self>::from_addr(pa);
207        pd.set(idx, PdEntry::present_leaf_with(leaf_flags, m2));
208    }
209}

211impl MapSize for Size4K {
212    fn ensure_chain_for<A: PhysFrameAlloc, M: PhysMapper>(
213        aspace: &AddressSpace<M>,
214        alloc: &mut A,
215        va: VirtualAddress,
216        nonleaf_flags: VirtualMemoryPageBits,
217    ) -> Result<PhysicalPage<Size4K>, MapSizeEnsureChainError> {
218        let i4 = L4Index::from(va);
219        let i3 = L3Index::from(va);
220        let i2 = L2Index::from(va);
221
222        // L4 → L3
223        let pml4 = aspace.pml4_mut();
224        let e4 = pml4.get(i4);
225        let pdpt_page = if let Some(p) = e4.next_table() {
226            p
227        } else {
228            let f = alloc.alloc_4k().ok_or(MapSizeEnsureChainError::OomPdpt)?;
229            aspace.zero_pdpt(f);
230            pml4.set(i4, Pml4Entry::present_with(nonleaf_flags, f));
231            f
232        };
233
234        // L3 → L2 (and split 1GiB if necessary)
235        let pdpt = aspace.pdpt_mut(pdpt_page);
236        let e3 = pdpt.get(i3);
237        let pd_page = match e3.kind() {
238            Some(PdptEntryKind::NextPageDirectory(pd, _)) => pd,
239            Some(PdptEntryKind::Leaf1GiB(_, _)) | None => {
240                let f = alloc.alloc_4k().ok_or(MapSizeEnsureChainError::OomPd)?;
241                aspace.zero_pd(f);
242                pdpt.set(i3, PdptEntry::present_next_with(nonleaf_flags, f));
243                f
244            }
245        };
246
247        // L2 → L1 (and split 2MiB if necessary)
248        let pd = aspace.pd_mut(pd_page);
249        let e2 = pd.get(i2);
250        Ok(match e2.kind() {
251            Some(PdEntryKind::NextPageTable(pt, _)) => pt,
252            Some(PdEntryKind::Leaf2MiB(_, _)) | None => {
253                let f = alloc.alloc_4k().ok_or(MapSizeEnsureChainError::OomPt)?;
254                aspace.zero_pt(f);
255                pd.set(i2, PdEntry::present_next_with(nonleaf_flags, f));
256                f
257            }
258        })
259    }
260
261    fn set_leaf<M: PhysMapper>(
262        aspace: &AddressSpace<M>,
263        leaf_tbl_page: PhysicalPage<Size4K>,
264        va: VirtualAddress,
265        pa: PhysicalAddress,
266        leaf_flags: VirtualMemoryPageBits,
267    ) {
268        debug_assert_eq!(pa.offset::<Self>().as_u64(), 0);
269        let pt = aspace.pt_mut(leaf_tbl_page);
270        let idx = L1Index::from(va);
271        let k4 = PhysicalPage::<Self>::from_addr(pa);
272
273        let entry = PtEntry4k::present_with(leaf_flags, k4);
274        pt.set(idx, entry);
275    }
