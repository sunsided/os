8/// Unified, ergonomic view over x86-64 paging entries (all levels / forms).
9///
10/// This type deliberately does **not** use bit-packing. Instead, it models the
11/// *semantic superset* of fields across:
12///
13/// - L4: [`Pml4Entry`] (non-leaf only)
14/// - L3: [`Pdpte`] (non-leaf) / [`Pdpte1G`] (1 GiB leaf)
15/// - L2: [`Pde`] (non-leaf) / [`Pde2M`]  (2 MiB leaf)
16/// - L1: [`PtEntry4k`]  (4 KiB leaf)
17///
18/// Use the provided `from_*` and `to_*` helpers to map between this view and
19/// the actual bitfield entries. Alignment and level-specific constraints are
20/// validated in `to_*` conversions (debug assertions).
21#[derive(Clone, Copy, Debug, Default, Eq, PartialEq, Setters)]
22#[allow(clippy::struct_excessive_bools)]
23pub struct VirtualMemoryPageBits {
24    /// **Present** (P): valid entry if `true`.
25    pub present: bool,
26
27    /// **Writable** (RW): allows writes if `true`.
28    pub writable: bool,
29
30    /// **User/Supervisor** (US): user-mode access if `true`.
31    pub user: bool,
32
33    /// **Page Write-Through** (PWT): write-through policy (PAT bit0).
34    pub write_through: bool,
35
36    /// **Page Cache Disable** (PCD): disable caching (PAT bit1).
37    pub cache_disable: bool,
38
39    /// **Accessed** (A): set by CPU on first access. Not a permission bit.
40    pub accessed: bool,
41
42    /// **Dirty** (D): set by CPU on first write.
43    ///
44    /// Meaningful for **leaf** entries only. Ignored in non-leaf mappings.
45    pub dirty: bool,
46
47    /// **Global** (G): TLB not flushed on CR3 reload (leaf only).
48    pub global: bool,
49
50    /// **Execute Disable** / NX: disallow instruction fetch if `true`.
51    ///
52    /// Requires `EFER.NXE` support. Applies across the walk (permission
53    /// intersection).
54    pub no_execute: bool,
55
56    /// **Protection key** (PKU) or OS-use (bits 59..62).
57    ///
58    /// Only meaningful if PKU is supported; otherwise free for OS use.
59    /// Value is masked to 4 bits on `to_*`.
60    pub protection_key: u8,
61
62    /// **OS-available (low)**: bits 9..11 (masked to 3 bits in `to_*`).
63    pub os_available_low: u8,
64
65    /// **OS-available (high)**: bits 52..58 (masked to 7 bits in `to_*`).
66    pub os_available_high: u8,
67
68    /// PAT selector **bit #2**:
69    ///
70    /// - **4 KiB leaf**: this is PTE bit7 (PAT small).
71    /// - **2 MiB leaf**: this is PDE bit12 (PAT large).
72    /// - **1 GiB leaf**: this is PDPTE bit12 (PAT large).
73    /// - **Non-leaf**: **ignored** and forced to `false` in `to_*`.
74    ///
75    /// Combined with `write_through`(bit3) and `cache_disable`(bit4) to form the
76    /// 3-bit PAT index: `[pat_bit2 : PCD : PWT]`.
77    pub pat_bit2: bool,
78}

212    /// Encode into [`Pml4Entry`] (non-leaf).
213    ///
214    /// - Requires `form == L4Entry`
215    /// - Enforces 4 KiB alignment; ignores `dirty`, `global`, `pat_bit2`.
216    #[must_use]
217    pub const fn to_pml4e(&self, page: PhysicalPage<Size4K>) -> Pml4Entry {
218        let mut e = Pml4Entry::new();
219        e.set_present(self.present);
220        e.set_writable(self.writable);
221        e.set_user(self.user);
222        e.set_write_through(self.write_through);
223        e.set_cache_disable(self.cache_disable);
224        e.set_accessed(self.accessed);
225        e.set_no_execute(self.no_execute);
226        e.set_os_available_low(self.os_available_low & 0b111);
227        e.set_os_available_high(self.os_available_high & 0x7F);
228        e.set_protection_key(self.protection_key & 0x0F);
229        e.set_physical_address(page);
230        e
231    }
