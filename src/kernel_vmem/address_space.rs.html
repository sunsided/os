1//! # Address Space (x86-64, PML4-rooted)
2//!
3//! Strongly-typed helpers to build and manipulate a **single** virtual address
4//! space (tree rooted at a PML4). This complements the typed paging layers
5//! (`PageMapLevel4`, `PageDirectoryPointerTable`, `PageDirectory`, `PageTable`).
6//!
7//! ## Highlights
8//!
9//! - [`AddressSpace::map_one`] to install one mapping (4 KiB / 2 MiB / 1 GiB).
10//! - [`AddressSpace::unmap_one`] to clear a single 4 KiB PTE.
11//! - [`AddressSpace::query`] to translate a VA to PA (handles huge pages).
12//! - [`AddressSpace::activate`] to load CR3 with this space’s root.
13//!
14//! ## Design
15//!
16//! - Non-leaf entries are created with caller-provided **non-leaf flags**
17//!   (typically: present + writable, US as needed). Leaf flags come from the
18//!   mapping call. We never silently set US/GLOBAL/NX; the caller decides.
19//! - Uses [`PhysicalPage<Size4K>`] for page-table frames, and [`VirtualAddress`] /
20//!   [`PhysicalAddress`] for endpoints. Alignment is asserted via typed helpers.
21//! - Keeps `unsafe` confined to mapping a physical frame to a typed table
22//!   through the [`PhysMapper`].
23//!
24//! ## Safety
25//!
26//! - Mutating active mappings requires appropriate **TLB maintenance** (e.g.,
27//!   `invlpg` per page or CR3 reload).
28//! - The provided `PhysMapper` must yield **writable** references to table frames.
29

61    #[allow(clippy::missing_errors_doc)]
62    pub fn new(mapper: &'m M, alloc: &mut impl PhysFrameAlloc) -> Result<Self, AddressSpaceError> {
63        let pml4 = alloc.alloc_4k().ok_or(AddressSpaceError::OutOfMemory)?;
64        unsafe {
65            let table = mapper.phys_to_mut::<PageMapLevel4>(pml4.base());
66            *table = PageMapLevel4::zeroed();
67        }
68
69        let mut me = Self::from_root(mapper, pml4);
70
71        // Copy kernel half PML4 entries from current kernel PML4
72        let kern = unsafe { AddressSpace::from_current(mapper) };
73        me.clone_upper_half_from(&kern);
74        Ok(me)
75    }

77    /// View the **currently active** address space by reading CR3.
78    ///
79    /// # Safety
80    /// - Must run at CPL0 with paging enabled.
81    /// - Assumes CR3 points at a valid PML4 frame.

83    pub unsafe fn from_current(mapper: &'m M) -> Self {
84        let root_pa = unsafe { read_cr3_phys() };
85        let root = PhysicalPage::<Size4K>::from_addr(root_pa);
86        Self { root, mapper }
87    }

96    /// Load CR3 with this address space’s root.
97    ///
98    /// # Safety
99    /// You must ensure the CPU paging state (CR0/CR4/EFER) and code/data mappings
100    /// are consistent with the target space. Consider reloading CR3 or issuing
101    /// `invlpg` after changes to active mappings.
102    #[inline]
103    pub unsafe fn activate(&self) {
104        let cr3 = self.root.base().as_u64();
105        unsafe {
106            core::arch::asm!("mov cr3, {}", in(reg) cr3, options(nostack, preserves_flags));
107        }
108    }

125    /// Translate a `VirtualAddress` to `PhysicalAddress` if mapped.
126    ///
127    /// Handles 1 GiB and 2 MiB leaves by adding the appropriate **in-page offset**.
128    #[must_use]
129    pub fn query(&self, va: VirtualAddress) -> Option<PhysicalAddress> {
130        match self.walk(va) {
131            WalkResult::Leaf1G { base, .. } => {
132                let off = va.offset::<Size1G>();
133                Some(base.join(off))
134            }
135            WalkResult::Leaf2M { base, .. } => {
136                let off = va.offset::<Size2M>();
137                Some(base.join(off))
138            }
139            WalkResult::L1 { pte, .. } => {
140                let (base4k, _fl) = pte.page_4k()?;
141                let off = va.offset::<Size4K>();
142                Some(base4k.join(off))
143            }
144            WalkResult::Missing => None,
145        }
146    }

148    /// Map **one** page at `va → pa` with size `S` and `leaf_flags`.
149    ///
150    /// - Non-leaf links are created with `nonleaf_flags` (e.g., present+writable).
151    /// - Alignment is asserted (debug) via typed wrappers.
152    ///
153    /// # Errors
154    /// - An Out of Memory error occurred in one of the tables.
155    pub fn map_one<A: PhysFrameAlloc, S: MapSize>(
156        &self,
157        alloc: &mut A,
158        va: VirtualAddress,
159        pa: PhysicalAddress,
160        nonleaf_flags: VirtualMemoryPageBits,
161        leaf_flags: VirtualMemoryPageBits,
162    ) -> Result<(), AddressSpaceMapOneError> {
163        debug_assert_eq!(pa.offset::<S>().as_u64(), 0, "physical address not aligned");
164        let leaf_tbl = S::ensure_chain_for(self, alloc, va, nonleaf_flags).inspect_err(|err| {
165            warn!("physical address mapping error: {err:?}");
166        })?;
167
168        trace!("Mapped one {} page at VA={va} -> PA={pa}", S::as_str());
169        S::set_leaf(self, leaf_tbl, va, pa, leaf_flags);
170        Ok(())
171    }

173    /// Unmap a single **4 KiB** page at `va`. Returns Err if missing.
174    ///
175    /// # Errors
176    /// - Invalid tables
177    // TODO: Refactor to error type
178    pub fn unmap_one(&self, va: VirtualAddress) -> Result<(), &'static str> {
179        match self.walk(va) {
180            WalkResult::L1 { pt, i1, pte } => {
181                if !pte.present() {
182                    return Err("missing: pte");
183                }
184
185                trace!("Unmapped VA={va}");
186                pt.set_zero(i1);
187                Ok(())
188            }
189            WalkResult::Leaf2M { .. } => Err("found 2MiB leaf (not a 4KiB mapping)"),
190            WalkResult::Leaf1G { .. } => Err("found 1GiB leaf (not a 4KiB mapping)"),
191            WalkResult::Missing => Err("missing: chain"),
192        }
193    }

195    /// Greedy region mapping: tiles `[virt_start .. virt_start+len)` onto
196    /// `[phys_start .. phys_start+len)` using 1G / 2M / 4K pages as alignment permits.
197    ///
198    /// - Non-leaf links use `nonleaf_flags` (e.g. present|writable).
199    /// - Leaves use `leaf_flags` (e.g. perms, NX, GLOBAL).
200    ///
201    /// # Errors
202    /// - Propagates OOMs from intermediate table allocation.
203    pub fn map_region<A: PhysFrameAlloc>(
204        &self,
205        alloc: &mut A,
206        virt_start: VirtualAddress,
207        phys_start: PhysicalAddress,
208        len: u64,
209        nonleaf_flags: VirtualMemoryPageBits,
210        leaf_flags: VirtualMemoryPageBits,
211    ) -> Result<(), AddressSpaceMapRegionError> {
212        let mut off = 0u64;
213        while off < len {
214            let va = VirtualAddress::new(virt_start.as_u64() + off);
215            let pa = PhysicalAddress::new(phys_start.as_u64() + off);
216            let remain = len - off;
217
218            // Try 1 GiB
219            if (va.as_u64() & (Size1G::SIZE - 1) == 0)
220                && (pa.as_u64() & (Size1G::SIZE - 1) == 0)
221                && remain >= Size1G::SIZE
222            {
223                self.map_one::<A, Size1G>(alloc, va, pa, nonleaf_flags, leaf_flags)?;
224                off += Size1G::SIZE;
225                continue;
226            }
227
228            // Try 2 MiB
229            if (va.as_u64() & (Size2M::SIZE - 1) == 0)
230                && (pa.as_u64() & (Size2M::SIZE - 1) == 0)
231                && remain >= Size2M::SIZE
232            {
233                self.map_one::<A, Size2M>(alloc, va, pa, nonleaf_flags, leaf_flags)?;
234                off += Size2M::SIZE;
235                continue;
236            }
237
238            // Fall back to 4 KiB
239            if (va.as_u64() & (Size4K::SIZE - 1) == 0) && (pa.as_u64() & (Size4K::SIZE - 1) == 0) {
240                self.map_one::<A, Size4K>(alloc, va, pa, nonleaf_flags, leaf_flags)?;
241                off += Size4K::SIZE;
242                continue;
243            }
244
245            return Err(AddressSpaceMapRegionError::Unaligned(va, pa));
246        }
247        Ok(())
248    }

250    /// Greedy unmap of a region: clears whole 1G/2M leaves when aligned, otherwise 4K PTEs.
251    /// (Does not collapse tables; that's a separate optimization pass.)
252    pub fn unmap_region(&self, virt_start: VirtualAddress, len: u64) {
253        let mut off = 0u64;
254        while off < len {
255            let va = VirtualAddress::new(virt_start.as_u64() + off);
256            match self.walk(va) {
257                WalkResult::Leaf1G { pdpt, i3, .. }
258                    if (va.as_u64() & (Size1G::SIZE - 1) == 0) && (len - off) >= Size1G::SIZE =>
259                {
260                    // Clear the PDPTE
261                    pdpt.set_zero(i3);
262                    off += Size1G::SIZE;
263                }
264                WalkResult::Leaf2M { pd, i2, .. }
265                    if (va.as_u64() & (Size2M::SIZE - 1) == 0) && (len - off) >= Size2M::SIZE =>
266                {
267                    // Clear the PDE
268                    pd.set_zero(i2);
269                    off += Size2M::SIZE;
270                }
271                WalkResult::L1 { pt, i1, pte } => {
272                    if pte.present() {
273                        pt.set_zero(i1);
274                    }
275                    off += Size4K::SIZE;
276                }
277                // Missing entry: treat as unmapped 4K and advance to avoid infinite loop,
278                // or
279                // Leaf size larger than remaining or unaligned start: fall back to 4K step.
280                _ => off += Size4K::SIZE,
281            }
282        }
283    }

285    /// Walks the whole tree and frees empty tables (PT/PD/PDPT). Does not merge leaves.
286    #[allow(clippy::similar_names)]
287    pub fn collapse_empty_tables<F: PhysFrameAlloc>(&self, free: &mut F) {
288        let pml4 = self.pml4_mut();
289
290        // For every L4 entry:
291        for i4 in 0..512 {
292            let e4 = pml4.get(L4Index::new(i4));
293            let Some(pdpt_page) = e4.next_table() else {
294                continue;
295            };
296            let pdpt = self.pdpt_mut(pdpt_page);
297
298            // For every L3 entry:
299            let mut used_l3 = false;
300            for i3 in 0..512 {
301                match pdpt.get(L3Index::new(i3)).kind() {
302                    Some(PdptEntryKind::Leaf1GiB(_, _)) => {
303                        used_l3 = true;
304                    }
305                    Some(PdptEntryKind::NextPageDirectory(pd_page, _)) => {
306                        let pd = self.pd_mut(pd_page);
307                        // For every L2 entry:
308                        let mut used_l2 = false;
309                        for i2 in 0..512 {
310                            match pd.get(L2Index::new(i2)).kind() {
311                                Some(PdEntryKind::Leaf2MiB(_, _)) => {
312                                    used_l2 = true;
313                                }
314                                Some(PdEntryKind::NextPageTable(pt_page, _)) => {
315                                    let pt = self.pt_mut(pt_page);
316                                    let mut any_present = false;
317                                    for i1 in 0..512 {
318                                        if pt.get(L1Index::new(i1)).present() {
319                                            any_present = true;
320                                            break;
321                                        }
322                                    }
323                                    if any_present {
324                                        used_l2 = true;
325                                    } else {
326                                        // free PT
327                                        pd.set(L2Index::new(i2), PdEntry::zero());
328                                        free.free_4k(pt_page);
329                                    }
330                                }
331                                None => {}
332                            }
333                        }
334                        // If PD ended up empty (no leaves / no child PTs left), free it.
335                        if used_l2 {
336                            used_l3 = true;
337                        } else {
338                            pdpt.set(L3Index::new(i3), PdptEntry::zero());
339                            free.free_4k(pd_page);
340                        }
341                    }
342                    None => {}
343                }
344            }
345
346            // If PDPT is now empty, free it.
347            if !used_l3 {
348                pml4.set(L4Index::new(i4), Pml4Entry::zero());
349                free.free_4k(pdpt_page);
350            }
351        }
352    }

354    /// Internal walker: resolves VA to the point it terminates.
355    #[allow(clippy::similar_names)]
356    fn walk(&self, va: VirtualAddress) -> WalkResult<'_> {
357        let (i4, i3, i2, i1) = crate::page_table::split_indices(va);
358
359        // PML4
360        let pml4 = self.pml4_mut();
361        let Some(pdpt_page) = pml4.get(i4).next_table() else {
362            return WalkResult::Missing;
363        };
364
365        // PDPT
366        let pdpt = self.pdpt_mut(pdpt_page);
367        match pdpt.get(i3).kind() {
368            Some(PdptEntryKind::Leaf1GiB(base, _fl)) => WalkResult::Leaf1G { base, pdpt, i3 },
369            Some(PdptEntryKind::NextPageDirectory(pd_page, _fl)) => {
370                // PD
371                let pd = self.pd_mut(pd_page);
372                match pd.get(i2).kind() {
373                    Some(PdEntryKind::Leaf2MiB(base, _fl)) => WalkResult::Leaf2M { base, pd, i2 },
374                    Some(PdEntryKind::NextPageTable(pt_page, _fl)) => {
375                        // PT
376                        let pt = self.pt_mut(pt_page);
377                        let pte = pt.get(i1);
378                        WalkResult::L1 { pt, i1, pte }
379                    }
380                    None => WalkResult::Missing,
381                }
382            }
383            None => WalkResult::Missing,
384        }
385    }

435    /// Copy kernel upper-half PML4 entries (slots 256..=511) from `src` into `self`,
436    /// aliasing the same kernel page-table subtrees. Does not touch lower levels.
437    fn clone_upper_half_from(&mut self, src: &Self) {
438        let dst_pa = self.root_page();
439        let src_pa = src.root_page();
440
441        // Map both PML4 pages via HHDM
442        let dst_l4: &mut PageMapLevel4 = unsafe { self.mapper.phys_to_mut(dst_pa.base()) };
443        let src_l4: &mut PageMapLevel4 = unsafe { self.mapper.phys_to_mut(src_pa.base()) };
444
445        // Kernel half: indices 256..=511 (works for 48-bit and LA57)
446        for i in (256..512).map(L4Index::new) {
447            let e = src_l4.get(i);
448            if e.present() {
449                debug_assert!(!e.user(), "kernel PML4E must have US=0");
450            }
451            dst_l4.set(i, e);
452        }
453    }
454
455    /// Post-bringup clearing.
456    pub fn clear_lower_half(&mut self) {
457        let root = self.root_page();
458
459        // Map both PML4 pages via HHDM
460        let l4: &mut PageMapLevel4 = unsafe { self.mapper.phys_to_mut(root.base()) };
461
462        // User half: indices 0..256
463        for i in (0..256).map(L4Index::new) {
464            l4.set(i, Pml4Entry::zero());
465        }
466    }

476/// A mapping error.
477#[derive(Debug, Copy, Clone, PartialEq, Eq, thiserror::Error)]
478pub enum AddressSpaceMapRegionError {
479    #[error(transparent)]
480    OutOfMemory(#[from] MapSizeEnsureChainError),
481    #[error("unaligned va/pa for remaining size: {0:?} -> {1:?}")]
482    Unaligned(VirtualAddress, PhysicalAddress),
483}
484
485impl From<AddressSpaceMapOneError> for AddressSpaceMapRegionError {
486    fn from(e: AddressSpaceMapOneError) -> Self {
487        match e {
488            AddressSpaceMapOneError::OutOfMemory(e) => Self::OutOfMemory(e),
489        }
490    }
491}

493/// Target table/level produced by `ensure_chain`.
494#[derive(Copy, Clone, Debug, Eq, PartialEq)]
495pub enum EnsureTarget {
496    /// You will write a **PDPTE** (1 GiB leaf).
497    L3For1G,
498    /// You will write a **PDE** (2 MiB leaf).
499    L2For2M,
500    /// You will write a **PTE** (4 KiB leaf).
501    L1For4K,

504/// The result of a table walk.
505#[allow(dead_code)]
506enum WalkResult<'a> {
507    /// Hit a 1 GiB leaf at PDPT.
508    Leaf1G {
509        base: PhysicalPage<Size1G>,
510        pdpt: &'a mut PageDirectoryPointerTable,
511        i3: L3Index,
512    },
513    /// Hit a 2 MiB leaf at PD.
514    Leaf2M {
515        base: PhysicalPage<Size2M>,
516        pd: &'a mut PageDirectory,
517        i2: L2Index,
518    },
519    /// Reached PT (L1) with its index and current entry.
520    L1 {
521        pt: &'a mut PageTable,
522        i1: L1Index,
523        pte: PtEntry4k,
524    },
525    /// Missing somewhere in the chain.
526    Missing,
