1use crate::qemu_trace;
2use log::{LevelFilter, Log, Metadata, Record, SetLoggerError};
3
4pub struct QemuLogger {
5    max_level: LevelFilter,
6}
7
8impl QemuLogger {
9    #[must_use]
10    pub const fn new(max_level: LevelFilter) -> Self {
11        Self { max_level }
12    }
13
14    /// Call this once during early init.
15    #[allow(
16        static_mut_refs,
17        clippy::missing_errors_doc,
18        clippy::missing_panics_doc
19    )]
20    pub fn init(self) -> Result<(), SetLoggerError> {
21        // SAFETY: log::set_logger expects &'static Log. Use a leaked boxed (or a static) in kernels.
22        // For no-alloc, we'll use a static.
23        static mut LOGGER: Option<QemuLogger> = None;
24
25        // move self into static
26        unsafe {
27            LOGGER = Some(self);
28            // set_logger requires &'static dyn Log
29            log::set_logger(LOGGER.as_ref().unwrap() as &'static dyn Log)?;
30        }
31        log::set_max_level(LevelFilter::Trace);
32        Ok(())
33    }
34}
35
36impl Log for QemuLogger {
37    fn enabled(&self, metadata: &Metadata) -> bool {
38        metadata.level() <= self.max_level
39    }
40
41    fn log(&self, record: &Record) {
42        if !self.enabled(record.metadata()) {
43            return;
44        }
45
46        // Format: "[LEVEL] target: message\n"
47        // Keep allocations out â€” format directly into qemu_trace!
48        // qemu_trace! is assumed to accept format! style args.
49        qemu_trace!(
50            "[{}] {}: {}\n",
51            record.level(),
52            record.target(),
53            record.args()
54        );
55    }
56
57    fn flush(&self) {
58        // no-op for qemu debug port
59    }
60}
