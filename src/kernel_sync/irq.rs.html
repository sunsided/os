3/// A mutex guard that also disables interrupts while held.
4///
5/// `IrqMutex` combines an interrupt guard with a regular [`MutexGuard`].
6/// When created via [`Mutex::lock_irq`], it:
7///
8/// 1. saves the current interrupt state and disables interrupts, and
9/// 2. acquires the underlying mutex,
10///
11/// releasing them in reverse order on drop.
12///
13/// This prevents interrupt handlers from preempting the critical section
14/// and re-entering code that uses the same lock.
15///
16/// # Platform
17///
18/// Uses `cli/sti` and `pushfq/pop` and therefore targets `x86/x86_64`.
19///
20/// # Safety & Privilege
21///
22/// These operations must run in a context where `cli`/`sti` are legal
23/// (e.g., kernel or a suitable hypervisor context). Calling from user space
24/// or non-privileged modes is invalid.
25///
26/// # Examples
27///
28/// ```no_run
29/// use kernel_sync::{Mutex, RawSpin};
30/// use kernel_sync::irq::IrqMutex; // assuming this module
31///
32/// static M: Mutex<u64, RawSpin> = Mutex::from_raw(RawSpin::new(), 0);
33///
34/// // Disable interrupts and lock for the duration of the scope.
35/// {
36///     let _ig = M.lock_irq();
37///     // critical section guarded from both threads and interrupts
38/// }
39/// // interrupts and mutex are released here
40/// ```
41pub struct IrqMutex<'a, T, R: RawLock + RawUnlock> {
42    _irq: IrqGuard,
43    _g: MutexGuard<'a, T, R>,
44}
45
46impl<T, R: RawLock + RawUnlock> Mutex<T, R> {
47    /// Acquires the mutex with interrupts disabled for the guardâ€™s lifetime.
48    ///
49    /// This constructs an [`IrqGuard`] to save/disable interrupts, then
50    /// acquires the mutex and returns a paired [`IrqMutex`] guard. Dropping
51    /// the guard releases the mutex and restores interrupts if they were
52    /// previously enabled.
53    ///
54    /// # Platform / Privilege
55    ///
56    /// Requires `x86/x86_64` and a privileged execution context where
57    /// `cli/sti` are permitted.
58    #[inline]
59    pub fn lock_irq(&self) -> IrqMutex<'_, T, R> {
60        let ig = IrqGuard::new();
61        let g = self.lock();
62        IrqMutex { _irq: ig, _g: g }
63    }
64}

66/// Disables hardware interrupts (`cli`).
67///
68/// # Platform
69///
70/// `x86/x86_64`.
71///
72/// # Safety & Privilege
73///
74/// Must only be called in contexts where `cli` is permitted. Misuse can
75/// hang the system or violate execution environment rules.
76#[inline]
77pub fn cli_stop_interrupts() {
78    unsafe { core::arch::asm!("cli", options(nomem, nostack, preserves_flags)) }
79}
80
81/// Enables hardware interrupts (`sti`).
82///
83/// # Platform
84///
85/// `x86/x86_64`.
86///
87/// # Safety & Privilege
88///
89/// Must only be called in contexts where `sti` is permitted. Typically used
90/// to restore a previously disabled interrupt state.
91#[inline]
92pub fn sti_enable_interrupts() {
93    unsafe { core::arch::asm!("sti", options(nomem, nostack, preserves_flags)) }
94}

96/// Returns the current `RFLAGS` value (via `pushfq/pop`).
97///
98/// Bit 9 (`IF`) indicates whether interrupts are enabled.
99///
100/// # Platform
101///
102/// `x86/x86_64`.
103///
104/// # Safety & Privilege
105///
106/// Requires an execution context where reading flags this way is valid.
107#[inline]
108#[must_use]
109pub fn rflags() -> u64 {
110    let r: u64;
111    unsafe { core::arch::asm!("pushfq; pop {}", out(reg) r, options(nostack, preserves_flags)) }
112    r
113}

115/// RAII guard that disables interrupts on creation and restores them on drop.
116///
117/// `IrqGuard::new()` snapshots the `IF` bit (bit 9 of `RFLAGS`). If interrupts
118/// were enabled, it executes `cli`. On drop, it executes `sti` **only** if
119/// they were previously enabled, preserving the original state.
120///
121/// # Platform / Privilege
122///
123/// Requires `x86/x86_64` and a privileged context permitting `cli/sti`.
124///
125/// # Examples
126///
127/// ```no_run
128/// use kernel_sync::irq::{IrqGuard, rflags};
129///
130/// let before = rflags();
131/// {
132///     let _g = IrqGuard::new(); // interrupts disabled here if previously enabled
133///     // critical section
134/// }
135/// let after = rflags(); // IF restored to prior state
136/// ```
137pub struct IrqGuard {
138    /// Whether interrupts were enabled (IF=1) when the guard was created.
139    were_enabled: bool,
140}
141
142impl Default for IrqGuard {
143    fn default() -> Self {
144        Self::new()
145    }
146}
147
148impl IrqGuard {
149    /// Disables interrupts if they are currently enabled and remembers the state.
150    ///
151    /// Uses `rflags()` to read the IF bit and conditionally issues `cli`.
152    #[inline]
153    #[must_use]
154    pub fn new() -> Self {
155        let enabled = (rflags() & (1 << 9)) != 0;
156        if enabled {
157            cli_stop_interrupts();
158        }
159        Self {
160            were_enabled: enabled,
161        }
162    }
163}
164
165impl Drop for IrqGuard {
166    /// Restores interrupts (`sti`) only if they were previously enabled.
167    fn drop(&mut self) {
168        if self.were_enabled {
169            sti_enable_interrupts();
170        }
171    }
172}
