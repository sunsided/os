2use core::hint::spin_loop;
3use core::sync::atomic::{AtomicUsize, Ordering};

5/// A simple *ticket lock* implementation using atomics.
6///
7/// `RawTicket` provides a fair spinlock where threads acquire the lock
8/// in the order they requested it. Each thread receives a ticket number,
9/// and spins until its number matches the current owner.
10///
11/// Compared to [`RawSpin`](crate::RawSpin), this lock ensures fairness:
12/// threads cannot starve as tickets are served strictly in order.
13///
14/// This lock is intended for very short critical sections where spinning
15/// is cheaper than sleeping. It is not reentrant and does not support
16/// poisoning or ownership tracking.
17///
18/// # Examples
19///
20/// ```
21/// use kernel_sync::RawTicket;
22///
23/// let lock = RawTicket::new();
24///
25/// lock.lock();
26/// // critical section
27/// unsafe { lock.unlock(); }
28/// ```
29pub struct RawTicket {
30    /// The next available ticket number.
31    next: AtomicUsize,
32    /// The ticket number currently being served.
33    owner: AtomicUsize,
34}
35
36impl Default for RawTicket {
37    fn default() -> Self {
38        Self::new()
39    }
40}
41
42impl RawTicket {
43    /// Creates a new, unlocked ticket lock.
44    ///
45    /// # Examples
46    ///
47    /// ```
48    /// use kernel_sync::RawTicket;
49    ///
50    /// let lock = RawTicket::new();
51    /// ```
52    #[must_use]
53    pub const fn new() -> Self {
54        Self {
55            next: AtomicUsize::new(0),
56            owner: AtomicUsize::new(0),
57        }
58    }
59
60    /// Acquires the lock, spinning in FIFO order until available.
61    ///
62    /// Each caller obtains a ticket number and waits until that ticket
63    /// matches the current owner. This guarantees first-come, first-served
64    /// fairness among contending threads.
65    ///
66    /// While waiting, this function uses [`core::hint::spin_loop`]
67    /// to signal a busy-wait loop to the processor.
68    ///
69    /// # Blocking
70    ///
71    /// This method never sleeps or yields; it spins until the lock is free.
72    #[inline]
73    pub fn lock(&self) {
74        let ticket = self.next.fetch_add(1, Ordering::Relaxed);
75        // Acquire when we observe our turn
76        while self.owner.load(Ordering::Acquire) != ticket {
77            spin_loop();
78        }
79    }
80
81    /// Attempts to acquire the lock without blocking.
82    ///
83    /// Returns `true` if the lock was successfully acquired, or `false`
84    /// if another thread currently holds it.
85    ///
86    /// This method performs a fair check based on ticket order.
87    ///
88    /// # Examples
89    ///
90    /// ```
91    /// use kernel_sync::RawTicket;
92    ///
93    /// let lock = RawTicket::new();
94    /// assert!(lock.try_lock());
95    /// assert!(!lock.try_lock());
96    /// unsafe { lock.unlock(); }
97    /// ```
98    #[inline]
99    pub fn try_lock(&self) -> bool {
100        let owner = self.owner.load(Ordering::Relaxed);
101        let next = self.next.load(Ordering::Relaxed);
102        if next == owner {
103            // attempt to claim the next ticket
104            self.next
105                .compare_exchange(next, next + 1, Ordering::Acquire, Ordering::Relaxed)
106                .is_ok()
107        } else {
108            false
109        }
110    }
111
112    /// Releases the lock and advances ownership to the next ticket.
113    ///
114    /// # Safety
115    ///
116    /// This method must only be called if the current thread holds the lock.
117    /// Calling it without a matching successful [`lock`](Self::lock) or
118    /// [`try_lock`](Self::try_lock) leads to undefined behavior.
119    ///
120    /// # Examples
121    ///
122    /// ```
123    /// use kernel_sync::RawTicket;
124    ///
125    /// let lock = RawTicket::new();
126    /// lock.lock();
127    /// unsafe { lock.unlock(); }
128    /// ```
129    #[inline]
130    pub unsafe fn unlock(&self) {
131        // Release when we advance owner
132        let t = self.owner.load(Ordering::Relaxed);
133        self.owner.store(t + 1, Ordering::Release);
134    }
135}
136
137impl RawLock for RawTicket {
138    #[inline]
139    fn raw_lock(&self) {
140        self.lock();
141    }
142
143    #[inline]
144    fn raw_try_lock(&self) -> bool {
145        self.try_lock()
146    }
147}
148
149impl RawUnlock for RawTicket {
150    #[inline]
151    unsafe fn raw_unlock(&self) {
152        unsafe { self.unlock() }
153    }
154}
